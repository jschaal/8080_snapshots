'================================================================================
' Class:        clsLabels
' Purpose:      Manages assembler labels and simple "procedures" advertised on
'               the sheet so CPU code can resolve symbolic operands.
'               - ParseLabels:     builds maps of labels and address xrefs
'               - ParseProcedures: builds a set of callable procedure names
'               - GetLabel / GetByXref / AddLabel / IsProc helpers
'
' Label Types:
'   "A" = address (auto from the instruction grid)
'   "E" = EQU constant (added via EQU)
'   "D" = data row (string in OP1)
'
' Notes:        Documentation/formatting only. No functional changes.
' Source:       Stable snapshot (2026-02-12 18:38)
'================================================================================

Private pLabels As Object      ' name -> clsLabelRecord
Private pLabelXrefs As Object  ' addressHex -> clsLabelRecord
Private pProcedures As Object  ' name set for CALL proc dispatch
Private wsCPU As Worksheet

Private Sub Class_Initialize()
    Set wsCPU = ThisWorkbook.Worksheets("CPU")
    Set pLabels = CreateObject("Scripting.Dictionary")
    Set pLabelXrefs = CreateObject("Scripting.Dictionary")
    Set pProcedures = CreateObject("Scripting.Dictionary")

    ParseLabels
    ParseProcedures
End Sub

'------------------------------------------------------------------------------
' ParseLabels
' Stops after EMPTY_RUN_LIMIT consecutive rows where Label/OpCode/OP1/OP2 are all blank.
' This avoids reliance on formula-driven sentinels (e.g., "F") and makes parsing deterministic.
'------------------------------------------------------------------------------
Public Sub ParseLabels()
    Dim data As Variant
    Dim r As Long
    Dim labelRec As clsLabelRecord
    Dim line0 As Range
    Dim maxRows As Long
    Dim ofs_label As Long, ofs_opcode As Long, ofs_op1 As Long, ofs_op2 As Long
    Dim labelName As String, opcode As String, OP1 As String, op2 As String
    Dim addrDec As Long, addrHex As String
    Dim emptyRun As Long
    Const EMPTY_RUN_LIMIT As Long = 8   ' stop after 8 consecutive empty rows

    pLabels.RemoveAll
    pLabelXrefs.RemoveAll

    Set line0 = wsCPU.Range("Line0")

    ' Offsets into the CPU grid
    ofs_label = wsCPU.Range("ofs_label").value
    ofs_opcode = wsCPU.Range("ofs_opcode").value
    ofs_op1 = wsCPU.Range("ofs_op1").value
    ofs_op2 = wsCPU.Range("ofs_op2").value

    ' Upper bound so we don't scan forever: MemSize + 1 is perfect
    ' (Your grid is MemStart..MemEnd; MemSize is in hex on CPU sheet.)
    maxRows = CLng(usrHexToDec(CStr(wsCPU.Range("MemSize").value))) + 1
    If maxRows < 1 Then maxRows = 1
    If maxRows > 5000 Then maxRows = 5000 ' hard safety cap

    ' Pull a block once for speed: from column 1 through ofs_ops2
    data = line0.offset(0, 0).Resize(maxRows, ofs_op2 + 1).value

    emptyRun = 0

    For r = 1 To UBound(data, 1)

        ' Read the program fields from the array (1-based array)
        labelName = UCase$(Trim$(CStr(data(r, ofs_label + 1))))
        opcode = UCase$(Trim$(CStr(data(r, ofs_opcode + 1))))
        OP1 = UCase$(Trim$(CStr(data(r, ofs_op1 + 1))))
        op2 = Trim$(CStr(data(r, ofs_op2 + 1)))

        ' Determine whether this row is "program-empty"
        If (labelName = "") And (opcode = "") And (OP1 = "") And (op2 = "") Then
            emptyRun = emptyRun + 1
            If emptyRun >= EMPTY_RUN_LIMIT Then Exit For
            GoTo NextRow
        Else
            emptyRun = 0
        End If

        ' Only create a label record if there's a label text on this row
        If labelName <> "" Then

            Set labelRec = New clsLabelRecord

            ' Address is the Dec Line value at column 1 of the grid
            addrDec = CLng(data(r, 1))
            addrHex = Right$("0000" & hex$(addrDec), 4)

            labelRec.clsAddress = addrDec
            labelRec.clsAddressHex = addrHex

            ' Determine label type from opcode (no sheet calc)
            Select Case opcode
                Case "EQU"
                    labelRec.clsLabelType = "E"
                    ' EQU constant value stored in OP1 in your sheet model
                    labelRec.clsAddressHex = Right$("0000" & UCase$(OP1), 4)
                    labelRec.clsAddress = usrHexToDec(labelRec.clsAddressHex)
                    labelRec.clsSize = IIf(labelRec.clsAddress < 256, 8, 16)

                Case "DB"
                    labelRec.clsLabelType = "D"
                    ' Data payload stored in OP1 (string)
                    labelRec.clsData = CStr(data(r, ofs_op1 + 1))
                    labelRec.clsSize = Len(labelRec.clsData)

                Case Else
                    labelRec.clsLabelType = "A"
                    labelRec.clsSize = 16
            End Select

            ' Always add by label name (including EQU)
            If Not pLabels.Exists(labelName) Then
                pLabels.ADD labelName, labelRec
            End If

            ' Only add address xref for addressable rows (A, D)
            If (labelRec.clsLabelType = "A" Or labelRec.clsLabelType = "D") Then
                If Not pLabelXrefs.Exists(addrHex) Then
                    pLabelXrefs.ADD addrHex, labelRec
                End If
            End If

        End If

NextRow:
    Next r
End Sub

'------------------------------------------------------------------------------
' GetLabel
' Returns the label record if found, else Nothing.
'------------------------------------------------------------------------------
Public Function GetLabel(ByVal labelName As String) As clsLabelRecord
    If pLabels.Exists(labelName) Then
        Set GetLabel = pLabels(labelName)
    Else
        Set GetLabel = Nothing
    End If
End Function

'------------------------------------------------------------------------------
' AddLabel
' Adds a new label record (used by EQU). If already present, idempotent.
' Returns 0 on success, -1 on failure (empty input).
'------------------------------------------------------------------------------
Public Function AddLabel(ByVal labelName As String, ByVal LabelType As String, ByVal AddressHex As String)
    Dim labelRec As clsLabelRecord
    Dim res As Integer

    res = 0
    If labelName <> "" Then
        Set labelRec = New clsLabelRecord
        labelRec.clsAddress = usrHexToDec(AddressHex)
        labelRec.clsAddressHex = AddressHex

        If labelRec.clsAddress < 256 Then
            labelRec.clsSize = 8
        Else
            labelRec.clsSize = 16
        End If

        If Not pLabels.Exists(labelName) And Not pLabelXrefs.Exists(AddressHex) Then
            pLabels.ADD labelName, labelRec
            pLabelXrefs.ADD AddressHex, labelRec ' also allow lookup by address hex
        End If
    Else
        res = -1
    End If

    AddLabel = res
End Function

    End If
End Function

'------------------------------------------------------------------------------
' ParseProcedures
' Reads the "Procedures" column on CPU sheet into a dictionary (name set).
'------------------------------------------------------------------------------
Public Sub ParseProcedures()
    Dim data As Variant
    Dim r As Long
    Dim labelRec As clsLabelRecord
    Dim lastRow As Long
    Dim labelName As String

    pProcedures.RemoveAll

    With ThisWorkbook.Sheets("CPU")
        lastRow = .Cells(.Rows.Count, .Range("Procedures").Column).End(xlUp).row
        data = .Range(.Range("Procedures"), .Cells(lastRow, .Range("Procedures").Column)).value
    End With

    For r = 1 To UBound(data, 1)
        labelName = UCase(Trim(CStr(data(r, 1))))
        If labelName <> "" Then
            Set labelRec = New clsLabelRecord
            labelRec.clsLabelType = "P" ' procedure marker
            If Not pProcedures.Exists(labelName) Then
                pProcedures.ADD labelName, labelRec
            End If
        End If
    Next r
End Sub

'------------------------------------------------------------------------------
' IsProc
' Returns True if the name is a registered procedure (CALL shim).
'------------------------------------------------------------------------------
Public Function IsProc(ByVal labelName As String) As Boolean
    IsProc = pProcedures.Exists(labelName)
End Function

' ==============================================================================
' RunDynamicTests (DISCOVERY-BASED)
' Finds tests by scanning Column A on "Unit Tests" for any non-empty name.
' For each name, discovers the contiguous test rows starting at Hex Line = 0
' and ending at END/HLT or the first blank OpCode row, then runs/validates.
' ==============================================================================

Public Sub RunDynamicTests()
    Dim wsTest As Worksheet, wsEMU As Worksheet
    Dim memCapacity As Long
    Dim startCol As Integer, ofs_label As Integer
    Dim endCol As Integer
    Dim tests As Collection                    ' (TestName, Range, NameRow)
    Dim t As Variant
    Dim resultsCol As Long, runCol As Long, compileCol As Long
    Dim overrideStepMode As Boolean
    Dim overrideReset As Boolean
    Dim overrideTrace As Boolean
    Dim overrideSkipBreaks As Boolean
    Dim compileThis As Boolean

    Set wsTest = Sheets("Unit Tests")
    Set wsEMU = Sheets("CPU")

    ' Pull from CPU sheet (unchanged)
    memCapacity = usrHexToDec(wsEMU.Range("MemSize").value)
    ofs_label = wsEMU.Range("ofs_label").value
    startCol = ofs_label
    endCol = wsEMU.Range("ofs_op2").value
'  turn off step mode and force reset for testrunner
    overrideStepMode = Range("Step") = 1
    If overrideStepMode Then
        Range("Step") = 0
    End If
    overrideReset = Range("Reset") = 0
    If overrideReset Then
        Range("Reset") = 1
    End If
    overrideTrace = Range("Trace") = 1
    If overrideTrace Then
        Range("Trace") = 0
    End If
    overrideSkipBreaks = Range("SkipBreaks") = 0
    If overrideSkipBreaks Then
        Range("SkipBreaks") = 1
    End If
    
    ' Honor your Run flag column again
    runCol = wsTest.Range("RunTest").Column
    compileCol = wsTest.Range("CompileTest").Column
    resultsCol = wsTest.Range("TestRunner").Column

    Application.ScreenUpdating = False
    Application.Calculation = xlCalculationManual

    ' Discover tests by names in Column A
    Set tests = DiscoverAllTests(wsTest)

    Dim batchPassed As Boolean: batchPassed = True
    Dim passCount As Long: passCount = 0
    Dim totalTests As Long: totalTests = 0

    For Each t In tests
        Dim testName As String: testName = CStr(t(0))
        Dim testRange As Range: Set testRange = t(1)
        Dim nameRow As Long: nameRow = CLng(t(2))

        Dim runThis As Boolean
        runThis = (val(wsTest.Cells(nameRow, runCol).value) = 1)

        If runThis Then
            compileThis = (val(wsTest.Cells(nameRow, compileCol).value) = 1)
            If compileThis Then
                Assemble8080_ToMachine
            End If
            totalTests = totalTests + 1
            Application.StatusBar = "Currently Running: " & testName

            If Not testRange Is Nothing Then
                ' A) Clear emulator area
                wsEMU.Range("Line0").offset(0, ofs_label).Resize(memCapacity + 1, endCol).ClearContents
                wsEMU.Range("Line0").offset(0, startCol).Resize(memCapacity, 1).ClearContents

                ' B) Copy program rows:
                '    Source = 5 columns starting at Label (D..H) ? Dest = 4 columns (Label..OP2)
                wsEMU.Range("Line0").offset(0, ofs_label).Resize(testRange.Rows.Count, 4).value = _
                    testRange.Columns(3).Resize(, 5).value

    ' Clear console only if this test asserts console output
               If TestHasConsoleAssertions(testRange) Then
                   ClearConsole   ' clears UserForm AND ConsoleBuffer
               End If

                ' Debug breadcrumb
                'ConsolePrint "> copying " & testName & " rows=" & testRange.Rows.Count, True

                ' C) Execute
                wsEMU.Range("Reset").value = 1
                ConsolePrint ">" & testName & ":", False, False
                ResetAddressList
                Application.Calculate
                SelectEngine

                ' D) Validate
                If TestValidateMultipleCriteria(testRange) Then
                    wsTest.Cells(nameRow, resultsCol).value = "PASS"
                    wsTest.Cells(nameRow, resultsCol).Interior.Color = vbGreen
                    passCount = passCount + 1
                    Call SetRunFlagInControlTable(testName, 0)
                Else
                    wsTest.Cells(nameRow, resultsCol).value = "FAIL"
                    wsTest.Cells(nameRow, resultsCol).Interior.Color = vbRed
                    batchPassed = False
                End If

                ConsolePrint wsTest.Cells(nameRow, resultsCol).value, True
                'If Not stepMode Then DoEvents
            Else
                wsTest.Cells(nameRow, resultsCol).value = "SKIPPED"
                wsTest.Cells(nameRow, resultsCol).Interior.Color = RGB(200, 200, 200)
            End If
        Else
            wsTest.Cells(nameRow, resultsCol).value = "SKIPPED"
            wsTest.Cells(nameRow, resultsCol).Interior.Color = RGB(200, 200, 200)
        End If
    Next t

    Application.Calculation = xlCalculationAutomatic
    Application.Calculate
    Application.ScreenUpdating = True
    Application.StatusBar = False
    ConsolePrint "", True
    Range("consoleText") = usrForm.tbConsole.value

'  set step and reset mode to original selectsions
    If overrideStepMode Then
        Range("Step") = 1
    End If
    If overrideReset Then
        Range("Reset") = 0
    End If
    If overrideTrace Then
        Range("Trace") = 1
    End If
    If overrideSkipBreaks Then
        Range("SkipBreaks") = 0
    End If

    Dim finalMsg As String
    If batchPassed Then
        finalMsg = "OVERALL RESULT: PASS" & vbCrLf & _
                   "Passed " & passCount & " of " & totalTests & " tests."
        MsgBox finalMsg, vbInformation, "Test Runner Success"
    Else
        finalMsg = "OVERALL RESULT: FAIL" & vbCrLf & _
                   "Only " & passCount & " of " & totalTests & " tests passed."
        MsgBox finalMsg, vbCritical, "Test Runner Failure"
    End If
End Sub

' ==============================================================================
' DiscoverAllTests
' Scans Column A for non-empty names. For each, calls FindTestBlock to return
' the contiguous block of test rows (B..I columns), starting where Hex Line = 0.
' Returns a Collection of Variant triplets: (TestName, Range, NameRow)
' ==============================================================================
Private Function DiscoverAllTests(ws As Worksheet) As Collection
    Dim lastRowA As Long, r As Long
    Dim tests As New Collection
    Dim testName As String
    Dim block As Range

    lastRowA = ws.Cells(ws.Rows.Count, "A").End(xlUp).row

    For r = 1 To lastRowA
        testName = Trim$(CStr(ws.Cells(r, "A").value))
        ' Only consider proper test names: Unit_* (prevents terminating on stray text)
        If Len(testName) > 0 And UCase$(testName) Like "UNIT_*" Then
            Set block = FindTestBlock(ws, r)
            Dim item As Variant
            ReDim item(0 To 2)
            item(0) = testName
            Set item(1) = block
            item(2) = r          ' row of the name for PASS/FAIL cell
            tests.ADD item
        End If
    Next r

    Set DiscoverAllTests = tests
End Function

' ==============================================================================
' FindTestBlock
' Given the row of a test name in Column A, locate the test rows:
'   - Find the first row below where Column B (Hex Line) is 0
'   - From there, include rows while:
'       * Column E (OpCode) is not empty, AND
'       * We haven't passed an END/HLT (inclusive – we include that row),
'     or until we hit the first blank data row.
' Builds and returns a Range spanning columns B..I for those rows.
' ==============================================================================
Private Function FindTestBlock(ws As Worksheet, nameRow As Long) As Range
    Dim lastRow As Long: lastRow = ws.Cells(ws.Rows.Count, "B").End(xlUp).row
    Dim r As Long, startRow As Long, endRow As Long

    ' 1) Start at first "Hex Line = 0" below the named row (skip header lines)
    startRow = 0
    For r = nameRow + 1 To lastRow
        If Trim$(CStr(ws.Cells(r, "B").value)) = "0" Then
            startRow = r
            Exit For
        End If
        ' If another proper test name appears first, abort this one
        If Len(Trim$(CStr(ws.Cells(r, "A").value))) > 0 And _
           UCase$(Trim$(CStr(ws.Cells(r, "A").value))) Like "UNIT_*" Then
            Exit For
        End If
    Next r
    If startRow = 0 Then
        Set FindTestBlock = Nothing
        Exit Function
    End If

    ' 2) Scan forward until the next proper test name or sheet end
    '    DO NOT stop at END/HLT; include SUBs and DB data after those.
    endRow = startRow
    For r = startRow To lastRow
        If r > startRow Then
            Dim aTxt As String
            aTxt = UCase$(Trim$(CStr(ws.Cells(r, "A").value)))
            If Len(aTxt) > 0 And aTxt Like "UNIT_*" Then Exit For
        End If
        endRow = r
    Next r

    ' 3) Return B..I span (your test block columns)
    If endRow >= startRow Then
        Set FindTestBlock = ws.Range(ws.Cells(startRow, "B"), ws.Cells(endRow, "I"))
    Else
        Set FindTestBlock = Nothing
    End If
End Function

'------------------------------------------------------------------------------
' TestValidateMultipleCriteria
' Each row may assert an expected register/flag or Console substring.
' "ExpectedReg" & "ExpectedVal" named ranges point to the columns in the test.
'------------------------------------------------------------------------------
Private Function TestValidateMultipleCriteria(testSource As Range) As Boolean
    Dim r As Long
    Dim regName As String, expectedVal As String, actualVal As String
    Dim allPassed As Boolean: allPassed = True
    Dim useRegs As Boolean
    Dim colExpectedReg As Integer, colExpectedVal As Integer

    colExpectedReg = Range("ExpectedReg").Column - 1
    colExpectedVal = Range("ExpectedVal").Column - 1

    For r = 1 To testSource.Rows.Count
        regName = Trim(testSource.Cells(r, colExpectedReg).value)
        expectedVal = Trim(testSource.Cells(r, colExpectedVal).value)

        If regName <> "" Then
            If UCase$(regName) = "CONSOLE" Then
                allPassed = TestConsoleContains(expectedVal)
            Else
                useRegs = InStr("[ApBpCpDpEpFpHpLpIXIY]", regName)
                On Error Resume Next
                If useRegs Then
                    expectedVal = Right$("00" & expectedVal, 2)
                    actualVal = Right$("00" & Sheets("CPU").Range("R_" & regName).value, 2)
                Else
                    actualVal = Sheets("CPU").Range(regName).value
                End If
                On Error GoTo 0

                If UCase$(actualVal) <> UCase$(expectedVal) Then
                    allPassed = False
                End If
            End If
        End If

        If Not allPassed Then
            Exit For
        End If
    Next r

    TestValidateMultipleCriteria = allPassed
End Function

'------------------------------------------------------------------------------
' TestConsoleContains
' True if the form console contains "needle" (case-insensitive by default).
'------------------------------------------------------------------------------
Public Function TestConsoleContains(ByVal needle As String, Optional ByVal caseSensitive As Boolean = False) As Boolean
    Dim txt As String
    txt = ConsoleBuffer_Text()   ' <-- authoritative, synchronous source

    If caseSensitive Then
        TestConsoleContains = (InStr(1, txt, needle, vbBinaryCompare) > 0)
    Else
        TestConsoleContains = (InStr(1, txt, needle, vbTextCompare) > 0)
    End If
End Function

'------------------------------------------------------------------------------
' TestTestConsoleContainsInOrder
' True if all parts appear in the given order (not necessarily adjacent).
'------------------------------------------------------------------------------
Public Function TestTestConsoleContainsInOrder(ParamArray parts() As Variant) As Boolean
    Dim txt As String, pos As Long, i As Long, p As String, found As Long
    txt = ConsoleBuffer_Text()   ' <-- authoritative source

    pos = 1
    For i = LBound(parts) To UBound(parts)
        p = CStr(parts(i))
        found = InStr(pos, txt, p, vbTextCompare)
        If found = 0 Then
            TestTestConsoleContainsInOrder = False
            Exit Function
        End If
        pos = found + Len(p)
    Next i

    TestTestConsoleContainsInOrder = True
End Function

' Returns True if the test block contains any "Console" assertions
Private Function TestHasConsoleAssertions(testSource As Range) As Boolean
    Dim r As Long
    Dim colExpectedReg As Integer
    
    ' Your runner already uses these named ranges; we reuse them.
    colExpectedReg = Range("ExpectedReg").Column - 1   ' Align to testSource columns

    For r = 1 To testSource.Rows.Count
        If UCase$(Trim$(CStr(testSource.Cells(r, colExpectedReg).value))) = "CONSOLE" Then
            TestHasConsoleAssertions = True
            Exit Function
        End If
    Next r

    TestHasConsoleAssertions = False
End Function
' Updates the Run column (R) in the control table P5:R… for the given TestName.
' valueToSet: 0 or 1 (we’ll use 0 on PASS as requested).
Private Sub SetRunFlagInControlTable(ByVal testName As String, ByVal valueToSet As Long)
    Dim ws As Worksheet: Set ws = Sheets("Unit Tests")
    Dim firstCell As Range: Set firstCell = ws.Range("TestTable")   ' top-left of the control table (header row)
    
    ' Find the last used row of the control table based on Column P (TestName column).
    Dim lastRow As Long
    lastRow = ws.Cells(ws.Rows.Count, firstCell.Column).End(xlUp).row
    If lastRow < firstCell.row + 1 Then Exit Sub  ' no data rows

    ' Names range (Column P only)
    Dim rngNames As Range
    Set rngNames = ws.Range(firstCell.offset(1, 0), ws.Cells(lastRow, firstCell.Column))

    ' Find TestName inside Column P (exact match, case-insensitive by Excel default)
    Dim idx As Variant
    idx = Application.Match(testName, rngNames, 0)
    If IsError(idx) Then
        ' Not found ? nothing to do (keeps tests resilient if the name isn’t listed)
        Exit Sub
    End If

    ' Compute the row for Column R (Run column): that’s P + 2 columns
    Dim targetRow As Long
    targetRow = rngNames.row + CLng(idx) - 1

    ' Set Run (Column R) to the requested value (0 on PASS)
    ws.Cells(targetRow, firstCell.Column + 2).value = valueToSet
End Sub
Public Sub ResetTestCases()
    Dim ws As Worksheet: Set ws = Sheets("Unit Tests")
    Dim firstCell As Range: Set firstCell = ws.Range("TestTable")   ' top-left of the control table (header row)
    Dim lastRow As Long
    lastRow = ws.Cells(ws.Rows.Count, firstCell.Column + 1).End(xlUp).row
    If lastRow < firstCell.row + 1 Then Exit Sub  ' no data rows
    
    Dim rngNames As Range
    Set rngNames = ws.Range(firstCell.offset(1, 0), ws.Cells(lastRow, firstCell.Column))
    Dim idx As Range
    Dim targetRow As Long
    Dim testName As String
    Application.ScreenUpdating = False
    For Each idx In rngNames
    ' Compute the row for Column R (Run column): that’s P + 2 columns
        targetRow = idx.row
        testName = ws.Cells(targetRow, firstCell.Column + 1).value
        If testName <> "" Then
            ws.Cells(targetRow, firstCell.Column + 2).value = 1
        End If
    Next idx
    Application.ScreenUpdating = True

End Sub

    Me.tbConsole.SelStart = 0

Public Sub ConPrtLn(ByVal data As String, Optional ByVal prtSpace As Boolean = False)
    With Me.tbConsole
        .Locked = False
        ' Append text
        .value = .value & IIf(Len(.value) > 0 And prtSpace, " ", "") & data & vbCrLf
        
        ' Force Focus and Scroll

        .CurLine = .LineCount - 1
        
        .Locked = True
    End With

    Me.Repaint
    If Sheets("CPU").Range("Step").value = 1 Then DoEvents
End Sub

''' Component: Exportcode
Option Explicit
'================================================================================
' Module:       Exportcode
' Purpose:      Exports all VBA components to a single timestamped .txt file and
'               saves a binary snapshot (.xlsm) to one or more output folders.
'
' Requirements:
'   - Reference: Microsoft Visual Basic for Applications Extensibility 5.3
'   - Trust Center: "Trust access to the VBA project object model" (temporarily)
'
' Notes:        Uses late binding for VBIDE objects to reduce hard dependency.
'               Documentation/formatting only. No functional changes.
' Source:       Stable snapshot (2026-02-12 18:38)
'================================================================================

Public Sub ExportAllVbaToSingleText()
    Dim vbProj As Object ' late bound VBIDE.VBProject
    Dim vbComp As Object
    Dim codeMod As Object
    Dim f, g As Integer
    Dim outputDir As Variant
    Dim baseName As String
    Dim timeStamp As String
    Dim finalFileName As String
    Dim outputPath As String
    Dim outputPathStndard As String
    Dim fullPathXlsm As String
    Dim lastLine As Long
    Dim fso As Object

    ' 1) Timestamp
    timeStamp = Format(Now, "yyyy-mm-dd_HHmm")

    ' 2) Workbook name without extension
    baseName = ThisWorkbook.name
    If InStrRev(baseName, ".") > 0 Then
        baseName = Left(baseName, InStrRev(baseName, ".") - 1)
    End If

    ' 3) Compose consistent filename
    finalFileName = baseName & "_" & timeStamp
    

    ' 4) Output destinations (array of two folders)
    Dim base As Range, saveRow As Range
    Set base = Range("Exports")
    Set fso = CreateObject("Scripting.FileSystemObject")
    For Each saveRow In base.Rows
        If saveRow.Cells(1, 1) = 1 Then
            outputDir = saveRow.Cells(1, 2)
        ' Ensure trailing backslash
            If Right(outputDir, 1) <> "\" Then outputDir = outputDir & "\"

            ' Specific file paths
            outputPath = outputDir & finalFileName & ".txt"
            fullPathXlsm = outputDir & finalFileName & ".xlsm"
            outputPathStndard = outputDir & baseName & ".txt"

            ' Ensure folder exists
            On Error Resume Next
            If Not fso.FolderExists(outputDir) Then fso.CreateFolder outputDir
            On Error GoTo 0

            ' Open the text file
            f = FreeFile
            g = FreeFile
            Open outputPath For Output As #f
            
            g = FreeFile
            Open outputPathStndard For Output As #g

            Set vbProj = ThisWorkbook.VBProject
            Print #f, "=== VBA EXPORT: " & ThisWorkbook.name & " ==="
            Print #f, "Timestamp: " & Now
            Print #f, String(80, "=")
            
            Print #g, "=== VBA EXPORT: " & ThisWorkbook.name & " ==="
            Print #g, "Timestamp: " & Now
            Print #g, String(80, "=")
    
            For Each vbComp In vbProj.VBComponents
                Set codeMod = vbComp.CodeModule
                lastLine = codeMod.CountOfLines
    
                Print #f, vbCrLf & "''' Component: " & vbComp.name
                Print #g, vbCrLf & "''' Component: " & vbComp.name
                If lastLine > 0 Then
                    Print #f, codeMod.lines(1, lastLine)
                    Print #g, codeMod.lines(1, lastLine)
                Else
                    Print #f, "''' [No code]"
                    Print #g, "''' [No code]"
                End If
            Next vbComp

            Close #f
            Close #g

            ' Binary copy of workbook
            ThisWorkbook.SaveCopyAs fullPathXlsm
        End If
    Next saveRow

    MsgBox "Export complete for: " & finalFileName, vbInformation
End Sub

'------------------------------------------------------------------------------
' RunOpcode
' Routes a normalized opcode to its clsCPU implementation.
' Parameters:
'   myCpu  - CPU instance
'   opcode - already UCase/Trim by Execute (kept redundant for safety)
'   op1/op2 - operands (normalized by Execute)
'   curRow  - the line's Range for context (EQU label lookup)
' Returns:
'   0 on success; negative / nonzero error codes for fault/END
'------------------------------------------------------------------------------
Public Function RunOpcode(ByVal opcode As String, _
    ByVal OP1 As String, ByVal op2 As String, _
    Optional ByVal label As String = "") As Long

    Dim result As Long
    result = 0

    'opcode = Trim$(UCase$(opcode))
    Select Case opcode
        ' In Z80 machine code, CB is a prefix. In your row-based model,
        ' we treat the next row or the OP1/OP2 as the bit instruction.
        Case "CB": result = Me.CB(OP1)
        Case "ADD": result = Me.ADD(OP1)      ' 120xx
        Case "ADI": result = Me.ADI(OP1)      ' 130xx
        Case "ANA": result = Me.ANA(OP1)      ' 140xx
        Case "ANI": result = Me.ANI(OP1)      ' 150xx
        Case "CMP": result = Me.CMP(OP1)      ' 190xx
        Case "CPI": result = Me.CPI(OP1)      ' 200xx
        Case "CALL": result = Me.CALL_(OP1, op2)         ' 160xx error base inside
        Case "DB":   result = 0                              ' data pseudo-op
        Case "EQU":  result = Me.EQU(OP1, label)
        Case "END":  result = ERR_EXEC_END                ' soft termination
        Case "DCR":  result = Me.DCR(OP1)                 ' 230xx
        Case "HLT":  result = Me.HLT()                    ' -100
        Case "IN":   result = Me.IN_(OP1)                 ' 270xx
        Case "INR":  result = Me.INR(OP1)                 ' 280xx
        Case "JC":   result = Me.JC(OP1)
        Case "JM":   result = Me.JM(OP1)
        Case "JMP":  result = Me.JMP(OP1)                 ' 300xx
        Case "JNC":  result = Me.JNC(OP1)
        Case "JNZ":  result = Me.JNZ(OP1)
        Case "JP":   result = Me.JP(OP1)
        Case "JPE":  result = Me.JPE(OP1)
        Case "JPO":  result = Me.JPO(OP1)
        Case "JZ":   result = Me.JZ(OP1)
        Case "LXI":  result = Me.LXI(OP1, op2)            ' 340xx
        Case "MOV":  result = Me.MOV(OP1, op2)            ' 350xx
        Case "MVI":  result = Me.MVI(OP1, op2)            ' 360xx
        Case "NOP":  result = Me.NOP()
        Case "ORA", "OR":  result = Me.ORA(OP1)            ' 380xx
        Case "ORG":  result = Me.NOP()
        Case "ORI":  result = Me.ORI(OP1)                 ' 390xx
        Case "OUT":  result = Me.OUT_(OP1)                ' 400xx
        Case "POP":  result = Me.POP(OP1)                 ' 420xx
        Case "PUSH": result = Me.PUSH(OP1)                ' 430xx
        Case "RET":  result = Me.RET()                    ' 460xx
        
        ' --- Arithmetic with carry/borrow ---
        Case "ADC": result = Me.ADC(OP1)
        Case "ACI": result = Me.ACI(OP1)
        Case "SUB": result = Me.SUB_(OP1)
        Case "SUI": result = Me.SUI(OP1)
        Case "SBB": result = Me.SBB(OP1)
        Case "SBI": result = Me.SBI(OP1)
        
        ' --- Logical XOR ---
        Case "XRA": result = Me.XRA(OP1)
        Case "XRI": result = Me.XRI(OP1)
        
        ' --- Rotates / carry ops / complement / NOP ---
        Case "RLC": result = Me.RLC()
        Case "RRC": result = Me.RRC()
        Case "RAL": result = Me.RAL()
        Case "RAR": result = Me.RAR()
        Case "STC": result = Me.STC()
        Case "CMC": result = Me.CMC()
        Case "CMA": result = Me.CMA()
        
        ' --- 16-bit register-pair ops ---
        Case "INX": result = Me.INX(OP1)
        Case "DCX": result = Me.DCX(OP1)
        Case "DAD": result = Me.DAD(OP1)
        Case "DAA": result = Me.DAA()
        
        ' --- Register pair exchanges / flow helpers ---
        Case "XCHG": result = Me.XCHG()
        Case "SPHL": result = Me.SPHL()
        Case "PCHL": result = Me.PCHL()
        
        ' --- Conditional CALL ---
        Case "CZ":  result = Me.CCOND("Z", OP1)
        Case "CNZ": result = Me.CCOND("NZ", OP1)
        Case "CC":  result = Me.CCOND("C", OP1)
        Case "CNC": result = Me.CCOND("NC", OP1)
        Case "CP":  result = Me.CCOND("P", OP1)
        Case "CM":  result = Me.CCOND("M", OP1)
        Case "CPE": result = Me.CCOND("PE", OP1)
        Case "CPO": result = Me.CCOND("PO", OP1)
        
        ' --- Conditional RET ---
        Case "RZ":  result = Me.RCOND("Z")
        Case "RNZ": result = Me.RCOND("NZ")
        Case "RC":  result = Me.RCOND("C")
        Case "RNC": result = Me.RCOND("NC")
        Case "RP":  result = Me.RCOND("P")
        Case "RM":  result = Me.RCOND("M")
        Case "RPE": result = Me.RCOND("PE")
        Case "RPO": result = Me.RCOND("PO")
        
        ' --- address routines from
        Case "LDA":  result = Me.LDA(OP1)
        Case "STA":  result = Me.STA(OP1)
        Case "LHLD": result = Me.LHLD(OP1)
        Case "SHLD": result = Me.SHLD(OP1)
        Case "LDAX": result = Me.LDAX(OP1)
        Case "STAX": result = Me.STAX(OP1)
        Case "XTHL": result = Me.XTHL()
        ' --- IO ops (non standard 8080) ---
        Case "PRI": result = Me.PRI(OP1)

    End Select

    RunOpcode = result
End Function

'==============================================================================
'                       Instruction Implementations
'==============================================================================

' ==============================================================================
' ACI (Add Immediate with Carry to A)
' Usage: ACI data
' Opcode:   11001110 (next byte is immediate)
' 8080:     2 bytes, 7 cycles
' Z80:      2 bytes, 7 cycles
'
' Operation:   A ? A + immediate + CY
' Flags Affected:
'   CY = 1 if sum > 255 (overflow)
'   P  = 1 if result has even parity
'   AC = 1 if carry from bit 3
'   Z  = 1 if result is 0
'   S  = 1 if bit 7 is set
'   N  = 0 (Z80 only; indicates ADD, not SUB)
'
' Examples:
'   ACI 0x10        - Add 0x10 + carry to A
'   ACI 0xFF        - Add 0xFF + carry to A
'
' Common Uses:
'   - Multi-byte addition (add high byte with carry from low byte)
'   - Arithmetic with carry propagation
'
' Error Codes (base 100):
'   100: Missing operand (op1 is empty)
'   101: Invalid hex immediate (not valid hex or out-of-range)
' ==============================================================================
Public Function ACI(ByVal OP1 As String) As Long
    Dim errorBase As Long: errorBase = 100
    Dim imm8 As Variant

    imm8 = HlpResolveImm8(OP1, errorBase, "ACI")
    If IsEmpty(imm8) Then ACI = errorBase: Exit Function

    HlpALU_Addition CLng(imm8), True  ' ? WITH carry included!
    ACI = 0
End Function
' ---  ---
' End ACI

' ADC (Add Register with Carry to A)
' Usage: ADC source
' Opcode:   10001ss (ss = 0-7 for B,C,D,E,H,L,M,A)
' 8080:     1 byte, 4 cycles (7 if M)
' Z80:      1 byte, 4 cycles (7 if M)
'
' Operation:   A ? A + source + CY
' Flags Affected: Same as ADD (CY, P, AC, Z, S, N)
'
' Examples:
'   ADC B           - Add B + carry to A
'   ADC M           - Add memory at (HL) + carry to A
'
' Common Uses:
'   - Multi-byte addition
'   - 16-bit: First ADD low bytes, then ADC high bytes
'
' Error Codes (base 110):
'   110: Missing operand (op1 is empty)
'   111: Invalid register (op1 register name not recognized)
' ==============================================================================

    Dim errorBase As Long: errorBase = 110
    Dim src As Variant

    src = HlpResolveReg8(OP1, errorBase, "ADC")
    If IsEmpty(src) Then ADC = errorBase: Exit Function

    HlpALU_Addition CLng(src), True  ' True = include carry
    ADC = 0
End Function

' End ADC

    ANA = 0
End Function

' ==============================================================================
' ANI (AND Immediate with A)
' Usage: ANI data
' Opcode:   11100110 (next byte is immediate)
' 8080:     2 bytes, 7 cycles
' Z80:      2 bytes, 7 cycles
'
' Operation:   A ? A AND immediate
' Flags:       Same as ANA (Z, S, P; CY = 0, AC = 1)
'
' Examples:
'   ANI 0x0F        - Clear high nibble of A (mask with 0x0F)
'   ANI 0x80        - Test bit 7 (result 0 if not set)
'

' ==============================================================================
' CALL (Call Subroutine)
' Usage: CALL address
' Opcode:   11001101 (address follows as 16-bit word)
' 8080:     3 bytes, 17 cycles
' Z80:      3 bytes, 17 cycles
'
' Operation:   SP ? SP - 2
'              Memory[SP] ? next_instruction_address
'              PC ? address
' Flags Affected: NONE
'
' Examples:
'   CALL 0x1000     - Call subroutine at 0x1000
'   CALL 0x2000     - Call subroutine at 0x2000
'
' Common Uses:
'   - Call subroutines
'   - Function calls
'   - Modular code organization
'   - Reusable code blocks
'
' Error Codes (base 430):
'   430: Missing operand
'   431: Invalid address
' ==============================================================================
Public Function CALL_(ByVal OP1 As String, Optional ByVal op2 As String = "") As Long
    Dim record As clsLabelRecord, currentPC As Long, retAddr As Long
    Dim strAddr As String, sp_dec As Long, proc As Boolean
    Dim line0Address As Long
    Dim errorBase As Long: errorBase = 160
    Dim decAddr As Long
    Dim addrRec As clsAddrRecord

    If OP1 = "" Then
        SetError errorBase, "CALL: Missing Operand: " & OP1
        CALL_ = errorBase: Exit Function
    End If

    proc = pLabels.IsProc(OP1)
    If proc Then
        CALL_ = CallByName(Me, OP1, VbMethod, op2)
        Exit Function
    End If

    Set record = pLabels.GetLabel(OP1)
    If Not record Is Nothing Then
        decAddr = record.clsAddress
    Else
        If Not usrValIsHexString(OP1) Then
            SetError errorBase + 1, "CALL: Invalid Address OP1: " & OP1
            CALL_ = errorBase + 1
            Exit Function
        End If
        decAddr = usrHexToDec(OP1)
    End If

    currentPC = pPC_dec
    sp_dec = pRegs("SP")
    Set addrRec = pAddressList.GetAddr(currentPC)
    retAddr = addrRec.clsAddressNext

    strAddr = Right$("0000" & usrHexToString(retAddr), 4)

    sp_dec = sp_dec - 1
    pStack.clsPUSH sp_dec, usrHexToDec(Left$(strAddr, 2)), "CALL " & OP1
    sp_dec = sp_dec - 1
    pStack.clsPUSH sp_dec, usrHexToDec(Right$(strAddr, 2)), "CALL " & OP1
    
    RegSet "SP", sp_dec
    mUIStackDirty = True
    
    
    Set addrRec = pAddressList.GetAddr(decAddr)
    pPC_dec = addrRec.clsAddressLast
    SetPC pPC_dec
    
    CALL_ = 0
End Function
' ---  ---
' End CALL

' ==============================================================================
' DAA (Decimal Adjust Accumulator)
' Usage: DAA
' Opcode:   00100111
' 8080:     1 byte, 4 cycles
' Z80:      1 byte, 4 cycles
'
' Operation:   Adjust A for Binary-Coded Decimal (BCD) after ADD/ADC
'
' Logic (8080):
'   If (A & 0x0F) > 9 or AC = 1: A ? A + 0x06
'   If (A >> 4) > 9 or CY = 1: A ? A + 0x60
'
' Notes:
'   - Used after ADD/ADC when operands are BCD digits (0-9)
'   - Corrects result into valid BCD format
'   - Example: 0x09 + 0x01 = 0x0A, DAA converts to 0x10 (BCD 10)
'
' Flags Affected:
'   Z, S, P updated for new value
'   CY may be set if high nibble overflow
'   AC preserved from original operation
'
' Common Uses:
'   - BCD arithmetic
'   - Decimal calculations
'   - Converting binary to BCD
'
' Error Codes: None
' ==============================================================================
Public Function DAA() As Long
    Dim errorBase As Long: errorBase = 210
    Dim a As Long, ADD As Long
    Dim cyIn As Long, cyOut As Long
    Dim lowAdj As Boolean
    Dim result As Long

    a = pRegs("A") And 255
    cyIn = pFlags("Carry") And 1

    ADD = 0
    lowAdj = False

    If ((a And &HF) > 9) Or (pFlags("AC") = 1) Then
        ADD = ADD + &H6
        lowAdj = True
    End If

    If (a > &H99) Or (cyIn = 1) Then
        ADD = ADD + &H60
    End If

    result = a + ADD

    cyOut = IIf((result > 255) Or (a > &H99) Or (cyIn = 1 And (ADD And &H60) <> 0), 1, 0)
    HlpFlagCalcCarry cyOut
    HlpFlagCalcAC IIf(lowAdj, 1, 0)

    result = result And 255
    RegSet "A", result
    HlpFlagCalcSign result: HlpFlagCalcZero result: HlpFlagCalcParity result

    DAA = 0
End Function
' ---  DAA ---

' Operation:   HL ? HL + regpair

Public Function DAD(ByVal rp As String) As Long
    Dim errorBase As Long: errorBase = 220
    Dim hl As Long, addv As Variant, sum As Long

    
    addv = HlpResolveRegPair(rp, errorBase, "DAD")

    DAD = 0

Public Function IN_(ByVal OP1 As String) As Long
    Dim errorBase As Long: errorBase = 270
    Dim portVal As Long
    Dim dec_port As Long

    If OP1 = "" Then SetError errorBase, "IN: Missing Port OP1": IN_ = errorBase: Exit Function

    If Not usrValIsHexString(OP1) Then
        SetError errorBase + 1, "IN: Invalid Hex: " & OP1

                IN_ = errorBase + 3
            End If
        Case Else
            SetError errorBase + 4, "IN: Unconfigured Port: " & OP1
            IN_ = errorBase + 4
            Exit Function
     End Select
     DoEvents
     IN_ = 0
End Function

' Opcode:   00pp0001 (pp = 0-3 for BC, DE, HL, SP)

' Z80:      3 bytes, 10 cycles

' Operation:   register_pair ? address (16-bit value)

'   - Initialize address pointers
'   - Load constants into register pairs
'
' Error Codes (base 290):
'   290: Missing register pair
'   291: Invalid register pair
'   292: Missing address
'   293: Invalid address format
' ==============================================================================
Public Function LXI(ByVal OP1 As String, op2 As String) As Long
    Dim errorBase As Long: errorBase = 340
    Dim record As clsLabelRecord
    Dim useLabel As Boolean
    Dim address As String
    Dim regLow As String, regHigh As String
    Dim decLow As Long, decHigh As Long
    Dim sp As Long, memStart As Long

    If OP1 = "" Or op2 = "" Then SetError errorBase, "LXI: Missing Operands: " & OP1 & " " & op2: LXI = errorBase: Exit Function
    If OP1 <> "B" And OP1 <> "D" And OP1 <> "H" And OP1 <> "SP" Then SetError errorBase + 1, "LXI: Invalid Register: " & OP1: LXI = errorBase + 1: Exit Function

    Set record = pLabels.GetLabel(op2)
    useLabel = Not record Is Nothing
    If useLabel Then op2 = record.clsAddressHex

    If Not usrValIsHexString(op2) Then SetError errorBase + 2, "LXI: Invalid Address: " & op2: LXI = errorBase + 2: Exit Function

    address = Right$("0000" & op2, 4)
    regHigh = Left$(address, 2)
    regLow = Right$(address, 2)
    decLow = usrHexToDec(regLow)
    decHigh = usrHexToDec(regHigh)

    Select Case OP1
        Case "B": RegSet "B", decHigh: RegSet "C", decLow
        Case "D": RegSet "D", decHigh: RegSet "E", decLow
        Case "H": RegSet "H", decHigh: RegSet "L", decLow
        Case "SP"
            Me.RegSet "SP", usrHexToDec(address)
            mUIStackDirty = True

    End Select
    
    LXI = 0
End Function

' ==============================================================================
' POP (Pop Register Pair from Stack)
' Usage: POP regpair
' Opcode:   11pp0001 (pp = 0-3 for BC, DE, HL, PSW)
' 8080:     1 byte, 10 cycles
' Z80:      1 byte, 10 cycles
'
' Operation:   reg_low ? Memory[SP], reg_high ? Memory[SP+1]
'              SP ? SP + 2
' Flags Affected: None (unless PSW, then all flags restored)
'
' Examples:
'   POP B           - Pop BC from stack
'   POP H           - Pop HL from stack
'   POP PSW         - Pop A and Flags from stack
'
' Common Uses:
'   - Restore register pairs from stack
'   - Return from function calls
'   - Restore saved state
'
' Error Codes (base 390):
'   390: Missing operand
'   391: Invalid register pair
'   392: Stack underflow (if applicable)
' ==============================================================================
Friend Function POP(ByVal OP1) As Long
    Dim res As Long, sp As Long
    Dim registerRecH As clsDecRegisterState, registerRecL As clsDecRegisterState
    Dim decLow As Long, decHigh As Long
    Dim errorBase As Long: errorBase = 420
    Dim flagByte As Byte

    res = 0
    If OP1 = "" Then SetError errorBase, "POP: Missing Operand: " & OP1: POP = errorBase: Exit Function

    sp = pRegs("SP")
    Set registerRecL = pStack.clsPOP: sp = sp + 1
    Set registerRecH = pStack.clsPOP: sp = sp + 1
    decLow = registerRecL.value
    decHigh = registerRecH.value

    Select Case OP1
        Case "B": RegSet "B", decHigh: RegSet "C", decLow
        Case "D": RegSet "D", decHigh: RegSet "E", decLow
        Case "H": RegSet "H", decHigh: RegSet "L", decLow
        Case "PSW"
            RegSet "A", decHigh
            flagByte = decLow And 255
            pFlags("Sign") = IIf((flagByte And 128) <> 0, 1, 0)
            pFlags("Zero") = IIf((flagByte And 64) <> 0, 1, 0)
            pFlags("AC") = IIf((flagByte And 16) <> 0, 1, 0)
            pFlags("Parity") = IIf((flagByte And 4) <> 0, 1, 0)
            pFlags("Flag") = 1
            pFlags("Carry") = IIf((flagByte And 1) <> 0, 1, 0)
            mUIFlagsDirty = True
        Case Else
            SetError errorBase + 1, "POP: Invalid Register: " & OP1: POP = errorBase + 1: Exit Function
    End Select

    RegSet "SP", sp
    
    POP = res
End Function

' ==============================================================================
' PUSH (Push Register Pair onto Stack)
' Usage: PUSH regpair
' Opcode:   11pp0101 (pp = 0-3 for BC, DE, HL, PSW)
' 8080:     1 byte, 11 cycles
' Z80:      1 byte, 11 cycles
'
' Operation:   SP ? SP - 2
'              Memory[SP+1] ? reg_high, Memory[SP] ? reg_low
' Flags Affected: NONE
'
' Examples:
'   PUSH B          - Push BC onto stack
'   PUSH H          - Push HL onto stack
'   PUSH PSW        - Push A and Flags onto stack
'
' Common Uses:
'   - Save register pairs on stack
'   - Preserve values during function calls
'   - Create temporary storage
'
' Note:
'   - PSW = A and Flags register together
'   - Decrements SP by 2
'
' Error Codes (base 380):
'   380: Missing operand
'   381: Invalid register pair
'   382: Stack overflow (if applicable)
' ==============================================================================
Public Function PUSH(ByVal OP1) As Long
    Dim sp As Long
    Dim sp_high As String, sp_low As String
    Dim errorBase As Long: errorBase = 430
    Dim flagByte As Byte

    If OP1 = "" Then SetError errorBase, "PUSH: Missing Operand: " & OP1: PUSH = errorBase: Exit Function

    sp = pRegs("SP")
    sp_high = sp - 1
    sp = sp - 1
    sp_low = sp
    
    Select Case OP1
        Case "B": pStack.clsPUSH sp_high, pRegs("B"), "B": pStack.clsPUSH sp_low, pRegs("C"), "C"
        Case "D": pStack.clsPUSH sp_high, pRegs("D"), "D": pStack.clsPUSH sp_low, pRegs("E"), "E"
        Case "H": pStack.clsPUSH sp_high, pRegs("H"), "H": pStack.clsPUSH sp_low, pRegs("L"), "L"
        Case "PSW"
            pStack.clsPUSH sp, pRegs("A"), "A"
            sp = sp - 1
            flagByte = 0
            If pFlags("Sign") = 1 Then flagByte = flagByte + 128
            If pFlags("Zero") = 1 Then flagByte = flagByte + 64
            If pFlags("AC") = 1 Then flagByte = flagByte + 16
            If pFlags("Parity") = 1 Then flagByte = flagByte + 4
            flagByte = flagByte + 2     ' reserved bit set
            If pFlags("Carry") = 1 Then flagByte = flagByte + 1
            
            pStack.clsPUSH sp, flagByte, "Flags"
        Case Else
            SetError errorBase + 1, "PUSH: Invalid Register: " & OP1: PUSH = errorBase + 1: Exit Function
    End Select

    RegSet "SP", sp
    mUIStackDirty = True
    
    PUSH = 0
End Function
' --- PUSH ---

' ==============================================================================
' RAL (Rotate A Left Through Carry)
' Usage: RAL
' Opcode:   00010111
' 8080:     1 byte, 4 cycles
' Z80:      1 byte, 4 cycles
'
' Operation:   A ? (A << 1) | CY_old, bit 7 ? CY
'
' Example:
'   Before:  A = 10110010, CY = 1
'   After:   A = 01100101, CY = 1  (bit 7 = 1)
'
' Notes:
'   - Rotate left; CY enters bit 0
'   - Old bit 7 goes to CY
'   - Effectively: A = (A * 2) | CY, with CY = (A >> 7)
'
' Flags Affected:
'   CY = old bit 7
'   All other flags unchanged
'
' Error Codes: None
' ==============================================================================
Public Function RAL() As Long
    Dim a As Long, msb As Long, cy As Long, result As Long
    a = pRegs("A") And 255
    cy = pFlags("Carry") And 1
    msb = IIf((a And &H80) <> 0, 1, 0)
    result = ((a * 2) And 255) Or cy
    RegSet "A", result
    HlpFlagCalcCarry msb
    RAL = 0
End Function

' ==============================================================================
' RAR (Rotate A Right Through Carry)
' Usage: RAR
' Opcode:   00011111
' 8080:     1 byte, 4 cycles
' Z80:      1 byte, 4 cycles
'
' Operation:   A ? (A >> 1) | (CY_old << 7), bit 0 ? CY
'
' Example:
'   Before:  A = 10110011, CY = 1
'   After:   A = 11011001, CY = 1  (bit 0 = 1)
'
' Notes:
'   - Rotate right; CY enters bit 7
'   - Old bit 0 goes to CY
'
' Flags Affected:
'   CY = old bit 0
'   All other flags unchanged
'
' Error Codes: None
' ==============================================================================
Public Function RAR() As Long
    Dim a As Long, lsb As Long, cy As Long, result As Long
    a = pRegs("A") And 255
    cy = pFlags("Carry") And 1
    lsb = a And 1
    result = ((a \ 2) And 127) Or (cy * 128)
    RegSet "A", result
    HlpFlagCalcCarry lsb
    RAR = 0
End Function

' ==============================================================================
' RET (Return from Subroutine)
' Usage: RET
' Opcode:   11001001
' 8080:     1 byte, 10 cycles
' Z80:      1 byte, 10 cycles
'
' Operation:   PC ? Memory[SP], SP ? SP + 2 (pop return address)
' Flags Affected: NONE
'
' Notes:
'   - Pops return address from stack
'   - Continues execution from return address
'   - Complements CALL instruction
'
' Examples:
'   RET             - Return to caller
'
' Common Uses:
'   - Exit from subroutines
'   - Return from function calls
'   - Standard function return
'
' Error Codes:
'   None (hardware manages stack)
' ==============================================================================
Public Function RET() As Long
    Dim stackRec As clsDecRegisterState
    Dim Count As Long, newPC As Long, sp_dec As Long
    Dim retAddress As String, retAddress_dec As Long
    Dim line0Address As Long
    Dim errorBase As Long: errorBase = 460
    Dim addressRec As clsAddrRecord

    Count = pStack.Count
    If Count = 0 Then SetError errorBase, "RET: No Address on Stack": RET = errorBase: Exit Function

    Set stackRec = pStack.clsPeek
    If Left$(stackRec.RegisterName, 4) <> "CALL" Then SetError errorBase + 1, "RET: Invalid Address on Stack: " & stackRec.RegisterName: RET = errorBase + 1: Exit Function

    sp_dec = pRegs("SP")
    Set stackRec = pStack.clsPOP
    sp_dec = sp_dec + 1
    retAddress_dec = stackRec.value
    retAddress = usrHexToString(retAddress_dec)
    
    Set stackRec = pStack.clsPOP
    sp_dec = sp_dec + 1
    retAddress = usrHexToString(stackRec.value) & retAddress

    Me.RegSet "SP", sp_dec
    mUIStackDirty = True

    Set addressRec = pAddressList.GetAddr(usrHexToDec(retAddress))
    newPC = addressRec.clsAddressLast
    SetPC newPC

    RET = 0
End Function
' ---  ---

' Usage: XTHL
' Opcode:   11100011
' 8080:     1 byte, 18 cycles
' Z80:      1 byte, 19 cycles
'
' Operation:   HL ? Memory[SP]
'   temp_L ? L, L ? Memory[SP], Memory[SP] ? temp_L
'   temp_H ? H, H ? Memory[SP+1], Memory[SP+1] ? temp_H

' ==============================================================================
' EX (Exchange Register Pairs)
' Usage: EX dest, source
' Variants:
'   EX DE, HL       - Swap DE with HL (opcode: EBh)
'   EX AF, AF'      - Swap A and F with shadows (opcode: 08h)
'   EX (SP), HL     - Swap (SP) with HL (1-byte stack operation)
'
' 8080:     Limited support (varies)
' Z80:      Full support (1-2 bytes, 4-19 cycles)
'
' Operation:
'   Exchanges values of two registers
'
' Examples:
'   EX DE, HL       - Swap DE and HL values
'   EX (SP), HL     - Pop HL from stack, push original HL to stack
' ==============================================================================
Public Function EX(ByVal OP1 As String, ByVal op2 As String) As Long
    Dim errorBase As Long: errorBase = 2000
    ' Case 1: EX AF, AF' (Swaps main Accumulator/Flags with Shadows)
    If OP1 = "AF" Then
        Me.EX_AF_AF_PRIME
        
    ' Case 2: EX DE, HL (Swaps DE and HL register pairs)
    ElseIf OP1 = "DE" And op2 = "HL" Then
        Dim tDE As Long: tDE = Me.RegGet16("DE")
        Me.RegSet16 "DE", Me.RegGet16("HL")
        Me.RegSet16 "HL", tDE
        
    ' Case 3: EX (SP), HL (Swaps HL with the top of the Stack)
    ElseIf OP1 = "(SP)" And op2 = "HL" Then
        Dim tSP As Long, tHL As Long
        tHL = Me.RegGet16("HL")
        ' Read 16-bit value from stack
        tSP = pMemory.addr(Me.RegGet16("SP")) + (CLng(pMemory.addr(Me.RegGet16("SP") + 1)) * 256)
        If HlpMemCheckError("EX", Me.RegGet16("SP"), errorBase) Then
            EX = errorBase
            Exit Function
        End If
        
        ' Swap them
        Me.RegSet16 "HL", tSP
        ' Write original HL back to stack
        pMemory.addr(Me.RegGet16("SP")) = tHL And &HFF
        pMemory.addr(Me.RegGet16("SP") + 1) = (tHL \ 256) And &HFF
    Else
        SetError errorBase + 1, "EX: Invalid register pair: " & OP1 & " - " & op2
        EX = errorBase + 1
        Exit Function
    End If
    EX = 0
End Function
' ---  ---

Public Function EX_AF_AF_PRIME() As Long
    Dim tempA As Variant, tempF As Variant
    
    ' Swap A
    tempA = pRegs("A")
    pRegs("A") = pRegs("Ap")
    pRegs("Ap") = tempA
    
    ' Swap Flags (Internal dictionary to Shadow and back)
    ' This requires a helper to pack/unpack the flags into a byte
    tempF = HlpFlagPack()
    HlpFlagUnpack pRegs("Fp")
    pRegs("Fp") = tempF
    
    mUIRegsDirty = True
    EX_AF_AF_PRIME = 0
End Function
' ---  ---

' ==============================================================================
' EXX (Exchange Primary and Shadow Registers)
' Usage: EXX
' Opcode:   11011001 (single byte)
' 8080:     NOT AVAILABLE
' Z80:      1 byte, 4 cycles
'
' Operation:
'   Swap (B,C) ? ? (B',C')
'   Swap (D,E) ? ? (D',E')
'   Swap (H,L) ? ? (H',L')
'   Note: A and F are NOT swapped; use separate EX AF, AF' for that
'
' Flags:       None affected (flags not swapped unless using EX AF, AF')
'
' Usage:
'   EXX             - Quickly swap register set for interrupt handler
'   EX AF, AF'      - Swap A and F with shadow versions
'
' Example:
'   Original:    B=0x42, C=0x99, B'=0x11, C'=0x22
'   After EXX:   B=0x11, C=0x22, B'=0x42, C'=0x99
' ==============================================================================
Public Function EXX() As Long
    Dim temp As Variant
    Dim pairs As Variant, p As Variant
    pairs = Array("B", "C", "D", "E", "H", "L")
    
    For Each p In pairs
        temp = pRegs(p)
        pRegs(p) = pRegs(p & "p")
        pRegs(p & "p") = temp
    Next p
    mUIRegsDirty = True
    EXX = 0
End Function

    If Not result Then

    End If
    mUIFlagsDirty = True
    mUIRegsDirty = True
    SRL = 0
End Function

' Evaluates jump conditions against flags
Private Function HlpEvaluateCondition(ByVal opcode As String) As Boolean
    Select Case opcode
        Case "JZ":  HlpEvaluateCondition = (pFlags("Zero") = 1)
        Case "JNZ": HlpEvaluateCondition = (pFlags("Zero") = 0)
        Case "JC":  HlpEvaluateCondition = (pFlags("Carry") = 1)
        Case "JNC": HlpEvaluateCondition = (pFlags("Carry") = 0)
        Case "JM":  HlpEvaluateCondition = (pFlags("Sign") = 1)
        Case "JP":  HlpEvaluateCondition = (pFlags("Sign") = 0)
        Case "JPE": HlpEvaluateCondition = (pFlags("Parity") = 1)
        Case "JPO": HlpEvaluateCondition = (pFlags("Parity") = 0)
    End Select
End Function
' ---  ---

Private Function HlpTestCondition(ByVal cond As String) As Boolean
    Select Case cond
        Case "Z":  HlpTestCondition = (pFlags("Zero") = 1)
        Case "NZ": HlpTestCondition = (pFlags("Zero") = 0)
        Case "C":  HlpTestCondition = (pFlags("Carry") = 1)
        Case "NC": HlpTestCondition = (pFlags("Carry") = 0)
        Case "M":  HlpTestCondition = (pFlags("Sign") = 1)
        Case "P":  HlpTestCondition = (pFlags("Sign") = 0)
        Case "PE": HlpTestCondition = (pFlags("Parity") = 1)
        Case "PO": HlpTestCondition = (pFlags("Parity") = 0)
    End Select

'------------------------------------------------------------------------------
' HlpFlagCalcParity (fast via LUT): sets Parity=1 for even parity as on 8080
'------------------------------------------------------------------------------
Private Function HlpFlagCalcParity(ByVal bits As Byte) As Long
    Static parity(0 To 255) As Byte
    Static init As Boolean
    Dim i As Long, b As Long, c As Long
    If Not init Then
        For i = 0 To 255
            b = i: c = 0
            Do While b
                c = c Xor (b And 1)
                b = b \ 2
            Loop
            ' 8080 parity flag: 1 for even parity
            parity(i) = IIf(c = 0, 1, 0)
        Next
        init = True
    End If
    pFlags("Parity") = parity(bits)
    mUIFlagsDirty = True
    HlpFlagCalcParity = pFlags("Parity")
End Function

' PRI
' Print Immediate OP1 value in either Hex or Ascii depending on A Register value
' op2 used to indicate whether to:
'   "X0": just print the byte
'   "X1": add space before after
'   "X2": add crlf after
'   "X3": add space before and crlf after
'   "0X": print ascii character
'   "1X": print hex character
Public Function PRI(ByVal OP1 As String) As Long
    Dim errorBase As Long: errorBase = 425
    Dim options As Long
    Dim prtSpace As Boolean, prtCrLf As Boolean, prtHex As Boolean
    Dim out As String
    Dim dec_op1 As Long
    Dim lbl As clsLabelRecord
    
    ' 1) Operand required
    If OP1 = "" Then
        SetError errorBase, "PRI: Missing Operand"
        PRI = errorBase
        Exit Function
    End If

    ' 2) Allow EQU labels as immediate (like your MVI does)
    Set lbl = pLabels.GetLabel(OP1)
    If Not lbl Is Nothing Then
        OP1 = lbl.clsAddressHex  ' use the label's hex value/address
    End If

    Dim op2 As Long
    op2 = pRegs("A")
    
    options = usrHexToDec(op2) And 255
    prtSpace = options And 1
    prtCrLf = options And 2
    prtHex = options And 16

    ' 3) Validate immediate (8-bit)
    If prtHex And Not usrValIsHexString(OP1) Then
        SetError errorBase + 1, "PRI: Invalid Hex Immediate: " & OP1
        PRI = errorBase + 1
        Exit Function
    End If
    
    If prtHex Then
        out = usrHexToAscii(OP1)
    Else
        out = OP1
    End If
    
    ConsolePrint out, prtCrLf, prtSpace
    PRI = 0
End Function

' PRT
' Prints literal string located at M (HL address pointer).  String is delimted by 0h end of line
' Reg A used to indicate whether to:
'   "X0": just print the byte
'   "X1": add space before after
'   "X2": add crlf after
'   "X3": add space before and crlf after
' prints its clsData. Optional CR/LF and space prefix handled by flags.
Public Function PRT() As Long
    Dim errorBase As Long: errorBase = 428
    Dim record As clsLabelRecord
    Dim address As String
    Dim address_high As String
    Dim address_low As String
    Dim dec_addr As Long
    Dim options As Long
    Dim prtSpace As Boolean, prtCrLf As Boolean
    Dim out As String
    Const MAX_STRING As Long = 1024   ' Maximum string length
    Dim outLen As Long: outLen = 0
    Dim outChar As Long
    
    options = pRegs("A") And 255
    prtSpace = options And 1
    prtCrLf = options And 2
    
    ' Resolve address in HL
    address_high = Right("00" & usrHexToString(pRegs("H")), 2)
    address_low = Right("00" & usrHexToString(pRegs("L")), 2)
    address = address_high & address_low
    
    out = ""
    dec_addr = usrHexToDec(address)
    outChar = pMemory.addr(dec_addr)
    If HlpMemCheckError("PRTN", dec_addr, errorBase) Then
        PRT = errorBase
        Exit Function
    End If
    
    Do While outChar <> 0 And outLen < MAX_STRING
        out = out & Chr(outChar)
        dec_addr = dec_addr + 1
        outChar = pMemory.addr(dec_addr)
        If HlpMemCheckError("PRTN", dec_addr, errorBase) Then
            PRT = errorBase
            Exit Function
        End If
        outLen = outLen + 1
    Loop
    
    ConsolePrint out, prtCrLf, prtSpace
    PRT = 0
End Function

                     Optional ByVal prtSpace As Boolean = False) As Long

    Dim record As clsLabelRecord

    PRTN = 0
End Function

Public Function GetInput(ByVal errorBase As Long) As Boolean
    Dim inputStr As String
    Dim addr As Long
    Dim addr_high As Long, addr_low As Long
    Dim i As Long
    'inputStr = InputBox("Enter Text:", "Console Input")
    inputStr = usrForm.tbInput.value
    If inputStr <> "" Then
        addr_high = pRegs("H")
        addr_low = pRegs("L")
        addr = addr_high * 256 + addr_low - 1
        For i = 1 To Len(inputStr)
            pMemory.addr(addr + i) = Asc(Mid(inputStr, i, 1))
            If HlpMemCheckError("IN", addr + i, errorBase) Then
                GetInput = False
                Exit Function
            End If
        Next i
        pMemory.addr(addr + Len(inputStr) + 1) = 0
        If HlpMemCheckError("IN", addr + Len(inputStr) + 1, errorBase) Then
            GetInput = False
            Exit Function
        End If
    End If
    GetInput = True
End Function

Private Function HlpMemAddrFromHL() As Long
    HlpMemAddrFromHL = ((pRegs("H") And 255) * 256&) + (pRegs("L") And 255)

Private Function HlpResolveReg8(ByVal operand As String, ByVal errorBase As Long, ByVal opname As String) As Variant
    operand = UCase$(Trim$(operand))
    If operand = "" Then
        SetError errorBase, opname & ": Missing operand"
        HlpResolveReg8 = Empty
        Exit Function
    End If

    If operand = "M" Then
        HlpResolveReg8 = CLng(pMemory.addr(HlpMemAddrFromHL())) And 255
        If HlpMemCheckError(opname, HlpMemAddrFromHL(), errorBase) Then
            HlpResolveReg8 = Empty
        End If
        Exit Function
    End If

    If pRegs.Exists(operand) Then
        HlpResolveReg8 = CLng(pRegs(operand)) And 255
        Exit Function
    End If

    SetError errorBase, opname & ": Invalid operand: " & operand
    HlpResolveReg8 = Empty
End Function

Private Function HlpStoreReg8(ByVal dest As String, ByVal value As Long, ByVal errorBase As Long, ByVal opname As String) As Boolean
    dest = UCase$(Trim$(dest))
    value = value And 255

    If dest = "M" Then
        pMemory.addr(HlpMemAddrFromHL()) = CByte(value)
        If HlpMemCheckError(opname, HlpMemAddrFromHL(), errorBase) Then
            HlpStoreReg8 = False
        Else
            HlpStoreReg8 = True
        End If
        Exit Function
    End If

    If pRegs.Exists(dest) Then
        RegSet dest, value
        HlpStoreReg8 = True
        Exit Function
    End If

    SetError errorBase, opname & ": Invalid destination: " & dest
    HlpStoreReg8 = False
End Function

' ==============================================================================
' BIT (Test Bit)
' Usage: BIT bit_pos, source
' Opcode:   CB 01bbsss (bb = bit 0-7, sss = source register)
' 8080:     NOT AVAILABLE
' Z80:      2 bytes, 8 cycles (12 if memory)
'
' Operation:   Test bit in source; set Zero flag if bit is 0
' Flags Affected:
'   Z = 1 if bit is 0 (bit not set)
'   Z = 0 if bit is 1 (bit is set)
'   S, P affected per result
'
' Examples:
'   BIT 0, A        - Test bit 0 of A (check if LSB is set)
'   BIT 7, B        - Test bit 7 of B (check sign bit)
'   BIT 3, M        - Test bit 3 of memory at (HL)
'
' Common Use:
'   BIT 0, A
'   JZ IS_EVEN      - Jump if bit 0 clear (even number)
' ==============================================================================
Public Function BIT(ByVal OP1 As String, ByVal op2 As String) As Long
    Dim errorBase As Long: errorBase = 2020
    Dim src As Variant
    Dim bitNum As Long
    
    If Not IsNumeric(OP1) Or val(OP1) < 0 Or val(OP1) > 7 Then
        SetError errorBase, "BIT: Invalid BIT " & OP1
    End If
    
    bitNum = CLng(val(OP1))
    src = HlpResolveReg8(op2, errorBase, "BIT")
    If IsEmpty(src) Then BIT = errorBase: Exit Function
    
    ' Set Zero flag if bit is 0
    pFlags("Zero") = IIf((src And (2 ^ bitNum)) = 0, 1, 0)
    pFlags("N") = 0        ' Z80 BIT resets N flag
    pFlags("AC") = 1       ' Z80 BIT sets Half-Carry flag
    mUIFlagsDirty = True
    BIT = 0
End Function
' ==============================================================================
' SET (Set Bit)
' SET_BIT bit_pos, dest
' Opcode:   CB 11bbddd (bb = bit 0-7, ddd = dest register)
' 8080:     NOT AVAILABLE
' Z80:      2 bytes, 8 cycles (15 if memory)
'
' Operation:   Set specified bit to 1 in destination
' Flags:       None affected
'
' Examples:
'   SET 7, A        - Set bit 7 of A (0x80 | A)
'   SET 0, C        - Set bit 0 of C
'   SET 4, M        - Set bit 4 of memory at (HL)
' ==============================================================================
Public Function SET_BIT(ByVal OP1 As String, ByVal op2 As String) As Long
    Dim bitNum As Long
    Dim errorBase As Long: errorBase = 2050
    
    If Not IsNumeric(OP1) Or val(OP1) < 0 Or val(OP1) > 7 Then
        SetError errorBase, "SET_BIT: Invalid BIT " & OP1
        SET_BIT = errorBase
        Exit Function
    End If
    
    bitNum = CLng(val(OP1))
    If Not pRegs.Exists(op2) Then
        SetError errorBase + 1, "SET_BIT: Invalid Register" & op2
        SET_BIT = errorBase + 1
        Exit Function
    End If
    pRegs(op2) = pRegs(op2) Or (2 ^ bitNum)
    mUIRegsDirty = True
    SET_BIT = 0
End Function
' ==============================================================================
' RES (Reset Bit)
' Usage: RES_BIT bit_pos, dest
' Opcode:   CB 10bbddd (bb = bit 0-7, ddd = dest register)
' 8080:     NOT AVAILABLE
' Z80:      2 bytes, 8 cycles (15 if memory)
'
' Operation:   Clear specified bit to 0 in destination
' Flags:       None affected
'
' Examples:
'   RES 7, A        - Clear bit 7 of A (A & 0x7F)
'   RES 0, H        - Clear bit 0 of H
' ==============================================================================
Public Function RES_BIT(ByVal OP1 As String, ByVal op2 As String) As Long
    Dim bitNum As Long
    Dim errorBase As Long: errorBase = 2050
    
    If Not IsNumeric(OP1) Or val(OP1) < 0 Or val(OP1) > 7 Then
        SetError errorBase, "RES_BIT: Invalid BIT " & OP1
        RES_BIT = errorBase
        Exit Function
    End If
    
    bitNum = CLng(val(OP1))
    If Not pRegs.Exists(op2) Then
        SetError errorBase + 1, "RES_BIT: Invalid Register" & op2
        RES_BIT = errorBase + 1
        Exit Function
    End If
    pRegs(op2) = pRegs(op2) And (Not (2 ^ bitNum))
    RES_BIT = 0
End Function
Public Function CB(ByVal OP1 As String) As Long
    Dim errorBase As Long: errorBase = 2030
    Dim result As Boolean
    ' Extract the sub-opcode from your OP1 column
    Dim subOp As Byte
    subOp = CByte(usrHexToDec(CStr(OP1)))
    ' Call the handler - passing the CPU object so it can modify registers
    result = HlpHandleCBPrefix(subOp, errorBase)
    If Not result Then
        CB = errorBase
        Exit Function
    End If
    CB = 0
End Function

' Helper to calculate (IX+d) or (IY+d)

    Dim d As Integer
    
    baseAddr = pRegs(regName)
    
    ' Convert hex displacement to signed integer
    d = CInt(usrHexToDec(CStr(displacement)))
    If d > 127 Then d = d - 256 ' Handle signed 8-bit wrap
    
    HlpMemGetIndexedAddr = (baseAddr + d) And &HFFFF&
End Function

        ' 40h - 7Fh are BIT b, r
        Case &H40 To &H7F
            ' Logic for BIT...
            
        ' 80h - BFh are RES b, r
        Case &H80 To &HBF
            ' Logic for RES...
            
        ' C0h - FFh are SET b, r
        Case &HC0 To &HFF
            ' Logic for SET...
    End Select
    HlpHandleCBPrefix = 0
End Function
' ==============================================================================
' RLC (Rotate A Left Circular)
' Usage: RLC
' Opcode:   00000111
' 8080:     1 byte, 4 cycles
' Z80:      1 byte, 4 cycles
'
' Operation:   A ? A << 1, bit 7 ? CY, CY ? bit 0
'
' Example:
'   Before:  A = 10110010, CY = 0
'   After:   A = 01100101, CY = 1  (bit 7 was 1)
'
' Flags Affected:
'   CY = old bit 7
'   All other flags unchanged
'
' Common Uses:
'   - Rotating bits left
'   - Extracting MSB
'   - Bit pattern manipulation
'
' Error Codes: None (no operands)
' ==============================================================================
' ==============================================================================
' SLA dest
' Opcode:   CB 00100ddd
' 8080:     NOT AVAILABLE
' Z80:      2 bytes, 8 cycles (15 if memory)
'
' Operation:
'   Shift left; bit 0 becomes 0; bit 7 moves to Carry
'   Before:  [bit7][bit6]...[bit1][bit0]
'   After:   [bit6]...[bit1][bit0][0] <- Carry gets old bit7
'
' Flags:       CY = old bit 7, Z,S,P per result
'
' Notes:
'   - Equivalent to multiply by 2
'   - Sign bit preserved implicitly (next SLA will rotate it out)
' ==============================================================================
' ==============================================================================
' SRL dest
' Opcode:   CB 00111ddd
' 8080:     NOT AVAILABLE
' Z80:      2 bytes, 8 cycles (15 if memory)
'
' Operation:
'   Shift right; bit 7 becomes 0; bit 0 moves to Carry
'   Before:  [bit7][bit6]...[bit1][bit0]
'   After:   [0][bit7]...[bit2][bit1] <- Carry gets old bit0
'
' Flags:       CY = old bit 0, Z,S,P per result
'
' Notes:
'   - Equivalent to unsigned divide by 2
' ==============================================================================
' ==============================================================================
' SRA dest
' Opcode:   CB 00101ddd
' 8080:     NOT AVAILABLE
' Z80:      2 bytes, 8 cycles (15 if memory)
'
' Operation:
'   Shift right; bit 7 preserved (sign-extending); bit 0 to Carry
'   Before:  [bit7][bit6]...[bit1][bit0]
'   After:   [bit7][bit7]...[bit2][bit1] <- Carry gets old bit0
'
' Flags:       CY = old bit 0, Z,S,P per result
'
' Notes:
'   - Preserves sign (arithmetic right shift)
'   - Equivalent to signed divide by 2
'   - Negative numbers shift in with 1s (preserved sign)
' ==============================================================================
Private Function HlpHandleCBShifts(ByVal op As Byte, ByVal opcode As String, ByVal errorBase As Long) As Boolean
    Dim regCode As Integer, opCodeType As Integer
    Dim regName As String
    Dim val As Byte, oldCarry As Byte, carryOut As Byte
    Dim res As Long ' Long to prevent overflow during bit shifts
    
    ' 1. Identify Register and Operation
    regCode = op And &H7          ' Bits 0-2: Register
    opCodeType = (op \ 8) And &H7 ' Bits 3-5: Operation
    
    ' 2. Get the value (Register or (HL))
    If regCode = 6 Then
        val = pMemory.addr(RegGet16("HL"))
        If HlpMemCheckError(opcode, Me.RegGet16("HL"), errorBase) Then
            HlpHandleCBShifts = False
            Exit Function
        End If
    Else
        regName = HlpRegGetName(regCode)
        val = pRegs(regName)
    End If
    
    oldCarry = pFlags("Carry")
    
    ' 3. Perform Operation
    Select Case opCodeType
        Case 0 ' RLC (Rotate Left Circular)
            carryOut = IIf((val And &H80) <> 0, 1, 0)
            res = ((val * 2) Or carryOut) And &HFF
            
        Case 1 ' RRC (Rotate Right Circular)
            carryOut = val And &H1
            res = (val \ 2) Or (carryOut * 128)
            
        Case 2 ' RL (Rotate Left through Carry)
            carryOut = IIf((val And &H80) <> 0, 1, 0)
            res = ((val * 2) Or oldCarry) And &HFF
            
        Case 3 ' RR (Rotate Right through Carry)
            carryOut = val And &H1
            res = (val \ 2) Or (oldCarry * 128)
            
        Case 4 ' SLA (Shift Left Arithmetic - Bit 0 becomes 0)
            carryOut = IIf((val And &H80) <> 0, 1, 0)
            res = (val * 2) And &HFF
            
        Case 5 ' SRA (Shift Right Arithmetic - Bit 7 is preserved)
            carryOut = val And &H1
            res = (val \ 2) Or (val And &H80)
            
        Case 6 ' SLL (Shift Left Logical - Bit 0 becomes 1, technically undocumented)
            carryOut = IIf((val And &H80) <> 0, 1, 0)
            res = ((val * 2) Or 1) And &HFF
            
        Case 7 ' SRL (Shift Right Logical - Bit 7 becomes 0)
            carryOut = val And &H1
            res = val \ 2
    End Select
    
    ' 4. Update Result
    If regCode = 6 Then
        pMemory.addr(RegGet16("HL")) = CByte(res)
    Else
        pRegs(regName) = CByte(res)
    End If
    
    ' 5. Update Z80 Flags
    pFlags("Carry") = carryOut
    pFlags("Zero") = IIf(res = 0, 1, 0)
    pFlags("Sign") = IIf((res And &H80) <> 0, 1, 0)
    pFlags("Parity") = HlpFlagCalcParity(CByte(res)) ' P/V stores parity for shifts
    pFlags("AC") = 0 ' Half-carry is always reset
    pFlags("N") = 0  ' Add/Subtract is always reset
    HlpHandleCBShifts = True
End Function
Private Function HlpRegGetName(ByVal code As Integer) As String
    Select Case code
        Case 0: HlpRegGetName = "B"
        Case 1: HlpRegGetName = "C"
        Case 2: HlpRegGetName = "D"
        Case 3: HlpRegGetName = "E"
        Case 4: HlpRegGetName = "H"
        Case 5: HlpRegGetName = "L"
        Case 7: HlpRegGetName = "A"
        Case Else: HlpRegGetName = "" ' 6 is (HL), handled separately
    End Select
End Function
Private Function HlpRegGetCode(ByVal reg As String) As Byte
    ' Standard Z80/8080 Register Mapping:
    ' B=000, C=001, D=010, E=011, H=100, L=101, (HL)=110, A=111
    
    Select Case UCase(Trim(reg))
        Case "B": HlpRegGetCode = 0
        Case "C": HlpRegGetCode = 1
        Case "D": HlpRegGetCode = 2
        Case "E": HlpRegGetCode = 3
        Case "H": HlpRegGetCode = 4
        Case "L": HlpRegGetCode = 5
        Case "(HL)", "M": HlpRegGetCode = 6 ' "M" is the 8080 name for (HL)
        Case "A": HlpRegGetCode = 7
        Case Else
            ' If the register isn't standard (like IX), we return 255 to flag an error
            HlpRegGetCode = 255
    End Select
End Function
Public Function RegGet16(ByVal regPair As String) As Long
    Dim result As Long
    
    ' Standardize input: remove spaces and handle different prime notations
    Dim key As String
    key = UCase(Trim(regPair))
    key = Replace(key, "PRIME", "'") ' Handle "BCPRIME" -> "BC'"
    
    Select Case key
        ' --- Standard Main Register Pairs ---
        Case "BC"
            result = (CLng(pRegs("B")) * 256) + pRegs("C")
        Case "DE"
            result = (CLng(pRegs("D")) * 256) + pRegs("E")
        Case "HL"
            result = (CLng(pRegs("H")) * 256) + pRegs("L")
        Case "AF"
            ' Uses your HlpFlagPack() to convert dictionary flags to a byte
            result = (CLng(pRegs("A")) * 256) + HlpFlagPack()

        ' --- Z80 Shadow (Prime) Pairs ---
        Case "BC'", "BC1"
            result = (CLng(pRegs("Bp")) * 256) + pRegs("C'")
        Case "DE'", "DE1"
            result = (CLng(pRegs("Dp")) * 256) + pRegs("E'")
        Case "HL'", "HL1"
            result = (CLng(pRegs("Hp")) * 256) + pRegs("L'")
        Case "AF'", "AF1"
            result = (CLng(pRegs("Ap")) * 256) + pRegs("F'")

        ' --- Index and Special Registers ---
        Case "IX"
            result = CLng(pRegs("IX"))
        Case "IY"
            result = CLng(pRegs("IY"))
        Case "SP"
            result = CLng(pRegs("SP"))
        Case "PC"
            result = CLng(pRegs("PC"))
            
        ' --- Special 8-bit combinations (I and R) ---
        Case "IR"
            result = (CLng(pRegs("I")) * 256) + pRegs("R")

        Case Else
            Debug.Print "Warning: RegGet16 unknown register: " & regPair
            result = 0
    End Select
    
    ' Apply the 16-bit mask to ensure no overflow and force positive Long
    RegGet16 = result And &HFFFF&
End Function
Public Sub RegSet16(ByVal regPair As String, ByVal value As Long)
    Dim val As Long
    val = value And &HFFFF& ' Safety mask
    
    Select Case UCase(Trim(regPair))
        Case "BC": pRegs("B") = (val \ 256): pRegs("C") = (val And &HFF)
        Case "DE": pRegs("D") = (val \ 256): pRegs("E") = (val And &HFF)
        Case "HL": pRegs("H") = (val \ 256): pRegs("L") = (val And &HFF)
        Case "AF"
            pRegs("A") = (val \ 256)
            HlpFlagUnpack CByte(val And &HFF) ' Updates pFlags dictionary
            
        Case "BC'": pRegs("Bp") = (val \ 256): pRegs("Cp") = (val And &HFF)
        Case "DE'": pRegs("Dp") = (val \ 256): pRegs("Ep") = (val And &HFF)
        Case "HL'": pRegs("Hp") = (val \ 256): pRegs("Lp") = (val And &HFF)
        Case "AF'": pRegs("Ap") = (val \ 256): pRegs("Fp") = (val And &HFF)
            
        Case "IX": pRegs("IX") = val
        Case "IY": pRegs("IY") = val
        Case "SP": pRegs("SP") = val
        Case "PC": pRegs("PC") = val
    End Select
    mUIFlagsDirty = True
End Sub
' ==============================================================================
' Indexed Addressing: (IX+d) and (IY+d)
'
' Purpose:    Access memory at (Index Register + signed displacement)
' Example:    LD A, (IX+5)     - Load A from memory at IX + 5
'             LD (IY-2), B     - Store B to memory at IY - 2
'
' Operand (d) is a signed 8-bit value (-128 to +127)
'
' 8080:       NOT AVAILABLE
' Z80:        Available with DD or FD prefix
'   DD = IX prefix
'   FD = IY prefix
'
' Execution:  address = Index_Reg + sign_extend(d)
'             Then perform operation at [address]
'
' Cycles:     Generally +4 cycles vs non-indexed variant
'             Example: LD A, (HL) = 7 cycles
'                      LD A, (IX+0) = 15 cycles (DD+opcode+displacement)
'
' Notes:
'   - Displacement is signed; allows negative offsets
'   - Essential for Z80 array/structure access
'   - Requires 2-byte opcode: DD/FD, then regular opcode
' ==============================================================================
Private Function LD(ByVal OP1 As String, ByVal op2 As String) As Long
    Dim errorBase As Long: errorBase = 2040
    On Error Resume Next
    Dim addr As Long
    ' 1. Indexed Target: LD (IX+5), A
    If InStr(OP1, "(") > 0 And (InStr(OP1, "IX") > 0 Or InStr(OP1, "IY") > 0) Then
        addr = HlpMemParseIndexAddr(OP1)
        pMemory.addr(addr) = HlpGetOperandValue(op2)
        If HlpMemCheckError("LD", addr, errorBase) Then
            LD = errorBase
            Exit Function
        End If
    ' 2. Indexed Source: LD A, (IX-1)
    ElseIf InStr(op2, "(") > 0 And (InStr(op2, "IX") > 0 Or InStr(op2, "IY") > 0) Then
        addr = HlpMemParseIndexAddr(op2)
        pRegs(OP1) = pMemory.addr(addr)
        If HlpMemCheckError("LD", addr, errorBase + 1) Then
            LD = errorBase + 1
            Exit Function
        End If
    ' 3. 16-Bit Register: LD HL, 1234H
    ElseIf Is16Bit(OP1) Then
        RegSet16 OP1, usrHexToDec(CStr(op2))
    ' 4. Standard 8-bit: LD A, B
    Else
        RegSet OP1, HlpGetOperandValue(op2)
    End If
    LD = 0
End Function
' Helper: Determines if the operand is a register or a hex literal
Private Function HlpGetOperandValue(ByVal op As String) As Byte
    op = Trim(UCase(op))
    If pRegs.Exists(op) Then
        ' It's a register (A, B, C, etc.)
        HlpGetOperandValue = CByte(pRegs(op) And &HFF)
    Else
        ' It's a hex value (e.g., 55H or 0xFF)
        HlpGetOperandValue = CByte(usrHexToDec(CStr(op)) And &HFF)
    End If
End Function

Private Function HlpMemParseIndexAddr(ByVal s As String) As Long
    ' Purpose: Converts strings like "(IX+05H)" or "(IY-1H)" into a 16-bit address
    ' s: The operand string from the Excel sheet
    
    On Error GoTo ParseError
    
    Dim cleanS As String
    cleanS = UCase(Replace(Replace(s, "(", ""), ")", "")) ' Remove parentheses
    
    Dim regPart As String
    Dim offsetPart As String
    Dim baseAddr As Long
    Dim offsetValue As Long
    
    ' Determine if we are using IX or IY
    regPart = Left(cleanS, 2) ' Should be IX or IY
    baseAddr = RegGet16(regPart)
    
    ' Check if there is an offset (+ or -)
    If InStr(cleanS, "+") > 0 Then
        offsetPart = Mid(cleanS, InStr(cleanS, "+") + 1)
        offsetValue = usrHexToDec(offsetPart)
    ElseIf InStr(cleanS, "-") > 0 Then
        offsetPart = Mid(cleanS, InStr(cleanS, "-") + 1)
        ' Ensure the offset is subtracted
        offsetValue = -usrHexToDec(offsetPart)
    Else
        ' No offset provided, e.g., "(IX)"
        offsetValue = 0
    End If
    
    ' Z80 addresses are 16-bit, so we use And &HFFFF& to wrap around
    HlpMemParseIndexAddr = (baseAddr + offsetValue) And &HFFFF&
    Exit Function

ParseError:
    Debug.Print "Error parsing Indexed Address: " & s
    HlpMemParseIndexAddr = 0
End Function

Private Function HlpMemCheckError(ByVal opcode As String, ByVal addr As Long, ByVal errorBase As Long) As Boolean
    If pMemory.MemError Then
        SetError errorBase, opcode & ": Invalid Memory Address: " & addr
        HlpMemCheckError = True
    Else
        HlpMemCheckError = False
    End If
End Function

'------------------------------------------------------------------------------
' HlpALU_Addition
' Purpose:      Perform 8-bit addition with automatic flag calculation
' Input:        operandValue - 8-bit value to add to A
'               includeCarry - If True, add carry flag to result
' Side Effects: Updates A, CY, AC, Z, S, P flags
'
' Flag Logic:
'   CY = 1 if sum > 255 (overflow)
'   AC = 1 if carry from bit 3 (nibble overflow)
'   Z  = 1 if result is 0
'   S  = 1 if bit 7 is set
'   P  = 1 if even parity
'   N  = 0 (addition, not subtraction)
'
' Used by: ADD, ADI, ADC, ACI
'------------------------------------------------------------------------------
Private Sub HlpALU_Addition(ByVal operandValue As Long, Optional ByVal includeCarry As Boolean = False)
    Dim a As Long, b As Long, sum As Long, result As Long
    Dim ac As Long, cy As Long
    
    ' Get current A register
    a = pRegs("A") And 255
    b = operandValue And 255
    
    ' Add carry if requested (ADC/ACI instructions)
    If includeCarry Then
        b = b + pFlags("Carry")
    End If
    
    ' Perform addition
    sum = a + b
    result = sum And 255
    
    ' Calculate flags
    cy = IIf(sum > 255, 1, 0)                           ' Carry from bit 7
    ac = IIf(((a And &HF) + (b And &HF)) > &HF, 1, 0)  ' Carry from bit 3
    
    ' Set flags
    pFlags("Carry") = cy
    HlpFlagCalcAC ac
    RegSet "A", result
    HlpFlagCalcSign result
    HlpFlagCalcZero result
    HlpFlagCalcParity result
    ' N flag = 0 (addition)
    If pFlags.Exists("N") Then pFlags("N") = 0
End Sub

'------------------------------------------------------------------------------
' HlpALU_Subtraction
' Purpose:      Perform 8-bit subtraction with automatic flag calculation
' Input:        operandValue - 8-bit value to subtract from A
'               includeBorrow - If True, subtract borrow flag from result
' Side Effects: Updates A, CY, AC, Z, S, P, N flags
'
' Flag Logic:
'   CY = 1 if A < operand (borrow)
'   AC = 1 if borrow from bit 3
'   Z  = 1 if result is 0
'   S  = 1 if bit 7 is set
'   P  = 1 if even parity
'   N  = 1 (subtraction operation, used for DAA)
'
' Used by: SUB, SUI, SBB, SBI
'------------------------------------------------------------------------------
Private Sub HlpALU_Subtraction(ByVal operandValue As Long, Optional ByVal includeBorrow As Boolean = False)
    Dim a As Long, b As Long, diff As Long, result As Long
    Dim borrow As Long, acBorrow As Long
    
    ' Get current A register
    a = pRegs("A") And 255
    b = operandValue And 255
    
    ' Subtract borrow if requested (SBB/SBI instructions)
    If includeBorrow Then
        b = b + pFlags("Carry")
    End If
    
    ' Calculate borrow before subtraction
    borrow = IIf(a < b, 1, 0)                        ' Borrow needed from bit 7
    acBorrow = IIf((a And &HF) < (b And &HF), 1, 0) ' Borrow from bit 3
    
    ' Perform subtraction (wrap-around for borrow)
    diff = a - b
    result = diff And 255  ' Mask to 8-bit (handles wrap-around)
    
    ' Set flags
    pFlags("Carry") = borrow           ' CY = 1 indicates borrow
    HlpFlagCalcAC acBorrow
    RegSet "A", result
    HlpFlagCalcSign result
    HlpFlagCalcZero result
    HlpFlagCalcParity result
    ' N flag = 1 (subtraction)
    If pFlags.Exists("N") Then pFlags("N") = 1
End Sub

'------------------------------------------------------------------------------
' HlpALU_AND
' Purpose:      Perform 8-bit logical AND with automatic flag calculation
' Input:        operandValue - 8-bit value to AND with A
' Side Effects: Updates A, Z, S, P flags
'               CY = 0, AC = 1 (Z80 behavior)
'
' Flag Logic:
'   A  = A AND operand
'   Z  = 1 if result is 0
'   S  = 1 if bit 7 is set
'   P  = 1 if even parity
'   CY = 0 (always cleared)
'   AC = 1 (always set, Z80 behavior)
'
' Used by: ANA, ANI
'------------------------------------------------------------------------------
Private Sub HlpALU_AND(ByVal operandValue As Long)
    Dim a As Long, b As Long, result As Long
    
    a = pRegs("A") And 255
    b = operandValue And 255
    result = a And b
    
    ' Set result and flags
    RegSet "A", result
    pFlags("Carry") = 0      ' CY = 0 (always)
    pFlags("AC") = 1         ' AC = 1 (Z80 behavior)
    HlpFlagCalcSign result
    HlpFlagCalcZero result
    HlpFlagCalcParity result
End Sub

'------------------------------------------------------------------------------

'   CY = 0 (always cleared)

'------------------------------------------------------------------------------
' HlpALU_Compare
' Purpose:      Perform 8-bit subtraction for comparison (discard result)
' Input:        operandValue - 8-bit value to compare with A
' Side Effects: Updates Z, S, P, CY, AC flags (A register unchanged)
'
' Flag Logic: Same as subtraction, but A is not modified
'   Z  = 1 if A == operand
'   S  = 1 if (A - operand) has bit 7 set
'   P  = 1 if even parity
'   CY = 1 if A < operand (borrow)
'   AC = 1 if borrow from bit 3
'
' Used by: CMP, CPI
'------------------------------------------------------------------------------
Private Sub HlpALU_Compare(ByVal operandValue As Long)
    Dim a As Long, b As Long, diff As Long
    Dim borrow As Long, acBorrow As Long
    
    a = pRegs("A") And 255
    b = operandValue And 255
    
    ' Calculate borrow before subtraction
    borrow = IIf(a < b, 1, 0)
    acBorrow = IIf((a And &HF) < (b And &HF), 1, 0)
    
    ' Perform subtraction (but don't save result to A)
    diff = a - b
    diff = diff And 255  ' Mask to 8-bit
    
    ' Set flags based on result (A unchanged)
    pFlags("Carry") = borrow
    HlpFlagCalcAC acBorrow
    HlpFlagCalcSign diff     ' Sign of result
    HlpFlagCalcZero diff     ' Zero flag from result
    HlpFlagCalcParity diff   ' Parity of result
End Sub

Public Sub InitHexTable()
    Dim i As Long ' Use Long to avoid overflow during loop
    For i = 0 To 255
        ' IMPORTANT: Use the built-in Hex$ function here, NOT usrHexToString
        HexTable(i) = Right$("0" & hex$(i), 2)
    Next i
    mIHexIintialized = True
End Sub
Public Function usrHexToString(ByVal value As Long) As String
    If Not mIHexIintialized Then InitHexTable
    
    ' Handle values as 16-bit if they exceed 255
    ' This covers PC, SP, and address operands automatically
    If value > 255 Then
        usrHexToString = HexTable(value \ 256) & HexTable(value Mod 256)
    ElseIf value < 0 Then
        ' Handle potential negative overflows if any
        usrHexToString = "0000"
    Else
        ' Standard 8-bit register (A, B, C, etc.)
        usrHexToString = HexTable(value)
    End If
End Function
'------------------------------------------------------------------------------
' usrValIsHexString
' Purpose:      Validate hex string input (with multiple format tolerances)
' Input:        hexValue - String to validate (may have leading/trailing space,
'                          trailing "H" suffix, or padding)
' Returns:      True if valid hex (1-4 digits, 0-9 A-F), False otherwise
' Examples:
'   usrValIsHexString("FF")      ? True
'   usrValIsHexString("0xFF")    ? True
'   usrValIsHexString(" FF ")    ? True
'   usrValIsHexString("FFH")     ? True
'   usrValIsHexString("F")       ? True (padded to 0F)
'   usrValIsHexString("FFFF")    ? True
'   usrValIsHexString("GG")      ? False (invalid hex digits)
'   usrValIsHexString("FFFFFF")  ? False (too long)
'
' Note: This consolidates usrIsValidHex() and the older validation logic
'       with better error handling and more flexible input tolerance.
'------------------------------------------------------------------------------
Public Function usrValIsHexString(ByVal hexValue As String) As Boolean
    Dim length As Long
    
    ' Step 1: Normalize input
    hexValue = Trim$(hexValue)                      ' Remove leading/trailing spaces
    hexValue = UCase$(hexValue)                     ' Convert to uppercase
    hexValue = Replace(hexValue, "H", "")           ' Remove trailing "H" if present
    
    ' Step 2: Check length
    length = Len(hexValue)
    
    ' Pad single or triple-digit hex to even length (2 or 4)
    If length = 1 Or length = 3 Then
        hexValue = "0" & hexValue
        length = length + 1
    End If
    
    ' Step 3: Validate length (must be 2 or 4 after padding)
    If length <> 2 And length <> 4 Then
        usrValIsHexString = False
        Exit Function
    End If
    
    ' Step 4: Validate hex digits (0-9, A-F only)
    ' Pattern: "*[!0-9a-fA-F]*" matches any character that is NOT hex
    If hexValue Like "*[!0-9A-F]*" Then
        usrValIsHexString = False
    Else
        usrValIsHexString = True
    End If
End Function

    Else

    End If
End Function

' z80 additions
' Helper: Simple check to see if we are dealing with 16-bit register logic
Public Function Is16Bit(ByVal r As String) As Boolean
    Select Case UCase(Trim(r))
        Case "BC", "DE", "HL", "IX", "IY", "SP", "AF", "BC'", "DE'", "HL'", "AF'": Is16Bit = True
        Case Else: Is16Bit = False
    End Select
End Function

Public Function HlpGetInstructionSize(ByVal opcode As String, Optional ByVal OP1 As String = "", Optional ByVal op2 As String = "") As Integer
    opcode = UCase(Trim(opcode))
    OP1 = UCase(Trim(OP1))
    op2 = UCase(Trim(op2))

    ' --- 4-Byte Instructions (Z80 Indexed + Immediate) ---
    ' Pattern: LD (IX+d), n
    If opcode = "LD" And (InStr(OP1, "IX") > 0 Or InStr(OP1, "IY") > 0) And IsNumeric(Left(op2, 1)) Then
        HlpGetInstructionSize = 4
        Exit Function
    End If
    ' Pattern: BIT/SET/RES on Indexed (Z80)
    If (opcode = "BIT" Or opcode = "SET" Or opcode = "RES") And (InStr(op2, "IX") > 0 Or InStr(op2, "IY") > 0) Then
        HlpGetInstructionSize = 4
        Exit Function
    End If

    ' --- 3-Byte Instructions (16-bit Immediates / Direct Addressing) ---
    Select Case opcode
        Case "JMP", "CALL", "JZ", "JNZ", "JC", "JNC", "JP", "JM", "JPE", "JPO", _
             "CZ", "CNZ", "CC", "CNC", "CP", "CM", "CPE", "CPO", _
             "STA", "LDA", "SHLD", "LHLD", "LXI"
            HlpGetInstructionSize = 3: Exit Function
            
        Case "LD"
            ' LXI BC, 1234H  or  LD BC, 1234H  or  LD IX, 1234H
            If Is16Bit(OP1) And Not IsRegister(op2) Then
                HlpGetInstructionSize = 3: Exit Function
            End If
            ' LD A, (1234H)
            If InStr(op2, "(") > 0 And Not (InStr(op2, "IX") > 0 Or InStr(op2, "IY") > 0 Or InStr(op2, "HL") > 0) Then
                HlpGetInstructionSize = 3: Exit Function
            End If
            
        Case "BIT", "SET", "RES", "SLA", "SRA", "SRL", "RLC", "RRC", "RL", "RR"
            ' Z80 Indexed version: LD (IX+d), r (this is a 3-byte op in machine code)
            If InStr(OP1, "IX") > 0 Or InStr(OP1, "IY") > 0 Or InStr(op2, "IX") > 0 Or InStr(op2, "IY") > 0 Then
                HlpGetInstructionSize = 3: Exit Function
            End If
        Case "EX"
            ' Check if we are using an Index Register (IX or IY)
            If InStr(OP1 & op2, "IX") > 0 Or InStr(OP1 & op2, "IY") > 0 Then
                HlpGetInstructionSize = 2
            Else
                HlpGetInstructionSize = 1
            End If
            Exit Function
    End Select

    ' --- 2-Byte Instructions (8-bit Immediates / Relative Jumps) ---
    Select Case opcode
        Case "MVI", "ADI", "ACI", "SUI", "SBI", "ANI", "XRI", "ORI", "CPI", "IN", "OUT", _
             "JR", "JRNZ", "JRZ", "JRC", "JRNC", "DJNZ"
            HlpGetInstructionSize = 2: Exit Function
            
        Case "LD"
            ' LD A, 55H (8-bit immediate)
            If Not Is16Bit(OP1) And Not IsRegister(op2) And InStr(op2, "(") = 0 Then
                HlpGetInstructionSize = 2: Exit Function
            End If
            
        Case "BIT", "SET", "RES", "SLA", "SRA", "SRL", "RL", "RR"
            ' Standard Z80 CB prefix instructions: BIT 7, A (2 bytes)
            HlpGetInstructionSize = 2: Exit Function
    End Select

    ' --- 1-Byte Instructions (Default) ---
    ' MOV, ADD, SUB, INR, DCR, PUSH, POP, RET, EX, EXX, NOP, "RLC", "RRC",  etc.
    HlpGetInstructionSize = 1
End Function

        Case "A", "B", "C", "D", "E", "H", "L", "M", "IX", "IY", "BC", "DE", "HL", "SP", "AF": IsRegister = True
        Case Else: IsRegister = False
    End Select

'------------------------------------------------------------------------------
' usrHexToDec
' Purpose:      Convert hex string to decimal value
' Input:        hexStr - Hex string (may have leading/trailing space, trailing "H")
'               Examples: "FF", "0xFF", " FF ", "FFH"
' Returns:      Long decimal value equivalent to hex input
' Errors:       Returns 0 if input is not valid hex (or uses On Error GoTo)
' Examples:
'   usrHexToDec("FF")       ? 255
'   usrHexToDec("0xFF")     ? 255
'   usrHexToDec(" FF ")     ? 255
'   usrHexToDec("FFH")      ? 255
'   usrHexToDec("1234")     ? 4660
'   usrHexToDec("ABCD")     ? 43981
'
' Note: Consolidates pusrHexToDec() with better error handling
'       and broader input tolerance.
'------------------------------------------------------------------------------
Public Function usrHexToDec(ByVal hexStr As String) As Long
    On Error GoTo ErrorHandler
    
    ' Step 1: Normalize input
    hexStr = Trim$(hexStr)                      ' Remove leading/trailing spaces
    hexStr = UCase$(hexStr)                     ' Convert to uppercase
    hexStr = Replace(hexStr, "H", "")           ' Remove trailing "H" if present
    
    ' Step 2: Convert hex string to decimal
    ' Val() with "&H" prefix tells VBA to interpret as hex
    usrHexToDec = val("&H" & hexStr)
    
    Exit Function
    
ErrorHandler:
    ' Return 0 on any error (invalid hex, empty string, etc.)
    usrHexToDec = 0
End Function
'------------------------------------------------------------------------------
' usrHexToAscii
' Returns '.' for non-printables and '?' for invalid hex.
'------------------------------------------------------------------------------
Public Function usrHexToAscii(ByVal hexRange As Variant) As String
    Dim hexVal As String
    Dim decVal As Integer

    If IsEmpty(hexRange) Or hexRange = "" Then
        usrHexToAscii = ""
        Exit Function
    End If

    hexVal = CStr(hexRange)
    On Error Resume Next
    decVal = Application.WorksheetFunction.Hex2Dec(hexVal)
    If Err.Number <> 0 Then
        usrHexToAscii = "?"
        Exit Function
    End If
    On Error GoTo 0

    If decVal >= 32 And decVal <= 126 Then
        usrHexToAscii = Chr(decVal)
    Else
        usrHexToAscii = "."
    End If
End Function

'------------------------------------------------------------------------------
' usrAsciiToHex
' Safe Ascii byte to Hex  string -> long (uses &H prefix trick)
'------------------------------------------------------------------------------
Public Function usrAsciiToHex(ByVal OP1 As String) As String
    If OP1 = "" Then
        usrAsciiToHex = "00"
        Exit Function
    End If
    Dim dec_char As Long
    dec_char = Asc(Left(OP1, 1))
    usrAsciiToHex = usrHexToString(dec_char)
End Function

'------------------------------------------------------------------------------
' ShowStack
' Renders the current stack (top at row 1) into named ranges on the CPU sheet.
' Skips redraw during headless runs unless force=True.
'------------------------------------------------------------------------------
Public Function ShowStack(Optional ByVal refresh As Boolean = False) As Boolean
    If Not refresh Then
        ShowStack = False
        Exit Function
    End If
    Dim i As Long, detail As Long
    Range("StackDetails").ClearContents

    If m_Count > 0 Then
        For i = 1 To m_Count
            detail = m_Count - i + 1
            Range("stackView").offset(i - 1, 1).value = usrHexToString(m_Items(detail).MemoryAddress)
            Range("stackView").offset(i - 1, 2).value = Right$("00" & usrHexToString(m_Items(detail).value), 2) & "H"
            Range("stackView").offset(i - 1, 3).value = m_Items(detail).RegisterName
        Next
    End If
    ShowStack = True
End Function

'------------------------------------------------------------------------------
' clsPUSH
' Pushes a new item with (address, value, register name) onto the stack.
'------------------------------------------------------------------------------
Public Sub clsPUSH(ByVal addr As Long, ByVal val As Long, ByVal reg As String)
    Dim newItem As clsDecRegisterState
    Set newItem = New clsDecRegisterState
    Dim memory As clsMemory
    
    Set memory = gMemory
    memory.addr(addr) = val
    
    newItem.MemoryAddress = addr
    newItem.value = val
    newItem.RegisterName = reg
    
    m_Count = m_Count + 1
    If m_Count = 1 Then
        ReDim m_Items(1 To 1)
    Else
        ReDim Preserve m_Items(1 To m_Count)
    End If

    Set m_Items(m_Count) = newItem
End Sub

'------------------------------------------------------------------------------
' clsPOP
' Pops the top item. Raises a descriptive error if underflow occurs.
'------------------------------------------------------------------------------
Public Function clsPOP() As clsDecRegisterState
    If m_Count > 0 Then
        Set clsPOP = m_Items(m_Count)
        m_Count = m_Count - 1
        If m_Count > 0 Then
            ReDim Preserve m_Items(1 To m_Count)
        Else
            ReDim m_Items(0)
        End If
    Else
        Err.Raise vbObjectError + 513, "clsStack", "Stack Underflow: No items to pop."
    End If
End Function

'------------------------------------------------------------------------------
' clsPeek
' Returns the top item without removing it. Nothing if empty.
'------------------------------------------------------------------------------
Public Function clsPeek() As clsDecRegisterState
    If m_Count > 0 Then
        Set clsPeek = m_Items(m_Count)
    Else
        Set clsPeek = Nothing
    End If
End Function

    Count = m_Count

'------------------------------------------------------------------------------
' Execute
' Entry point for a run. Honours "reset" and "Step" (headless) flags on the CPU
' sheet. Bulk-reads instruction columns to arrays for performance. Routes each
' opcode to RunOpcode (which dispatches into clsCPU).
'
' Sheet bindings (named ranges expected on "CPU"):
'   Start_pc_dec, pc_dec, reset, Error, errMessage
'   MemStart, MemSize, MemEnd
'   ofs_opcode, ofs_op1, ofs_op2, ofs_rowstat, ofs_label
'   Line0_dec (decimal first row of memory/instruction grid), StackStart, StackDetails,
'   consoleOut, Step
'------------------------------------------------------------------------------
Public Sub decExecute()
    Dim pc_dec As Long
    Dim opcode As String
    Dim OP1 As String, op2 As String, label As String
    Dim ofs_opcode As Long, ofs_op1 As Long, ofs_op2 As Long, ofs_rowstat As Long, ofs_label As Long
    Dim Reset As Long, errcode As Long
    Dim rowStat As String
    Dim memStart As Long, memEnd As Long
    Dim myCpu As clsDecCPU
    Dim headless As Boolean
    Dim prevCalc As XlCalculation
    Dim debugMode As Boolean
    Dim stepMode As Boolean
    Dim stepTaken As Boolean
    Dim executedThisIter As Boolean
    Dim endedByEnd As Boolean
    Dim skipBreaks As Boolean
    Dim traceReads As Boolean
    Dim maxIters As Long
    Dim stopRunning As Boolean

    ' --- Cached arrays mapped into locals for the run ---
    Dim base As Range, line0_dec As Long
    Dim countRows As Long
    Dim arrOpcode As Variant, arrRowStat As Variant, arrOp1 As Variant, arrOp2 As Variant, arrLabel As Variant
    Dim ubRows_Opcode As Long

    Dim rowIdx As Long

    Dim memAddrDec As Long
    Dim memBefore As String, memAfter As String, memAddrHex As String, memNote As String
    Dim memTrackLen As Long
    Dim preHL As Long, preSP As Long
    Dim b0 As Long, b1 As Long, a0 As Long, a1 As Long
    ' ========= Initialization =========
    InitHexTable

    ofs_opcode = Range("ofs_opcode").value
    ofs_op1 = Range("ofs_op1").value
    ofs_op2 = Range("ofs_op2").value
    ofs_rowstat = Range("ofs_rowstat").value
    ofs_label = Range("ofs_label").value

    debugMode = (Range("Debug").value = 1)
    stepMode = (Range("Step").value = 1)
    Reset = Range("Reset").value
    skipBreaks = Range("SkipBreaks") = 1
    traceReads = TraceMemReads()
    maxIters = Range("Max_Iters").value
    Range("Stop").value = 0

    errcode = 0
    endedByEnd = False
    stepTaken = False

    ' Clear sheet error/message each call (keeps step UI clean)
    Range("Error").value = 0
    Range("errMessage").value = ""

    ' check first and only first line for an org opcode to set global memory location
    If Range("Line0").offset(0, ofs_opcode) = "ORG" And Reset = 1 Then
        OP1 = Range("Line0").offset(0, ofs_op1).value
        If usrValIsHexString(OP1) Then
            Range("MemStart").value = OP1
            Application.Calculate
        End If
    End If
    
    memStart = usrHexToDec(Range("MemStart").value)
    memEnd = memStart + usrHexToDec(Range("MemSize").value)

    Set myCpu = gDecCPU                                         ' persistent CPU instance

    ' ========= Reset / continuation =========
    If Reset = 1 Then
        ' Fresh run / Step Start: reset CPU and rebuild cache
        If TraceEnabled() Then TraceClear
        ForceCompile (False)
        ResetAddressList
        ResetDecCPU
        Set myCpu = gDecCPU
        InvalidateCodeCache
        gBreak = False
        gCurrentIter = 0

        mConsoleOpen = False
        pc_dec = CLng(Range("Line0_dec").value)
        myCpu.SetPC pc_dec
        myCpu.reg("SP") = memEnd

        Range("StackStart").value = usrHexToString(myCpu.reg("SP"))
        Range("StackDetails").ClearContents
        If stepMode Then
            RefreshFlags myCpu
            RefreshRegisters myCpu
            myCpu.RefreshStack True
            Application.Calculate
        End If
    Else
        ' Continue run / Step: pull PC from persistent CPU object
        pc_dec = CLng(myCpu.reg("PC"))
    End If
    
    ' ========= Headless toggle =========
    headless = Not stepMode                                      ' stepMode implies interactive paints
    If headless Then
        Application.ScreenUpdating = False
        Application.EnableEvents = False
        prevCalc = Application.Calculation
        Application.Calculation = xlCalculationManual
    End If

    ' ========= Build/refresh cached arrays ONLY when needed =========
    Set base = Range("Line0_dec")
    line0_dec = CLng(base.value)

    countRows = (memEnd - memStart) + 1
    If countRows < 1 Then countRows = 1

    ' Rebuild cache only when Reset=1 or cache invalid or layout changed
    If (Not gCacheValid) _
       Or (gCacheCountRows <> countRows) _
       Or (gCacheLine0Dec <> line0_dec) _
       Or (gCacheMemStart <> memStart) _
       Or (gCacheMemEnd <> memEnd) _
       Or (gCacheOfsOpcode <> ofs_opcode) _
       Or (gCacheOfsOp1 <> ofs_op1) _
       Or (gCacheOfsOp2 <> ofs_op2) _
       Or (gCacheOfsRowStat <> ofs_rowstat) _
       Or (gCacheOfsLabel <> ofs_label) Then

        gCacheLine0Dec = line0_dec
        gCacheCountRows = countRows
        gCacheMemStart = memStart
        gCacheMemEnd = memEnd
        gCacheOfsOpcode = ofs_opcode
        gCacheOfsOp1 = ofs_op1
        gCacheOfsOp2 = ofs_op2
        gCacheOfsRowStat = ofs_rowstat
        gCacheOfsLabel = ofs_label

        gArrOpcode = base.offset(0, ofs_opcode).Resize(countRows, 1).value
        gArrRowStat = base.offset(0, ofs_rowstat).Resize(countRows, 1).value
        gArrOp1 = base.offset(0, ofs_op1).Resize(countRows, 1).value
        gArrOp2 = base.offset(0, ofs_op2).Resize(countRows, 1).value
        gArrLabel = base.offset(0, ofs_label).Resize(countRows, 1).value

        ' Normalize once (same intent as your existing loop)
        Dim idxN As Long
        For idxN = 1 To countRows
            If Not IsEmpty(gArrOpcode(idxN, 1)) Then gArrOpcode(idxN, 1) = Trim$(UCase$(CStr(gArrOpcode(idxN, 1))))
            If Not IsEmpty(gArrRowStat(idxN, 1)) Then gArrRowStat(idxN, 1) = Trim$(UCase$(CStr(gArrRowStat(idxN, 1))))
            If Not IsEmpty(gArrOp1(idxN, 1)) Then gArrOp1(idxN, 1) = Trim$(UCase$(CStr(gArrOp1(idxN, 1))))
            If Not IsEmpty(gArrOp2(idxN, 1)) Then gArrOp2(idxN, 1) = Trim$(CStr(gArrOp2(idxN, 1))) ' preserve string case for literals
            If Not IsEmpty(gArrLabel(idxN, 1)) Then gArrLabel(idxN, 1) = Trim$(UCase$(CStr(gArrLabel(idxN, 1))))
        Next idxN

        gCacheValid = True
    End If

    ' Use cached arrays for execution
    arrOpcode = gArrOpcode
    arrRowStat = gArrRowStat
    arrOp1 = gArrOp1
    arrOp2 = gArrOp2
    arrLabel = gArrLabel

    ubRows_Opcode = UBound(arrOpcode, 1)

    ' ========= Establish starting row index =========
    rowIdx = myCpu.GetCurrentIdx

    If rowIdx < 0 Or rowIdx >= ubRows_Opcode Then
        GoTo ExecFinalize
    End If

    opcode = CStr(arrOpcode(rowIdx + 1, 1))

    ' ========= Main loop =========
    Do While errcode = 0 And pc_dec <= memEnd And (Not stepMode Or (stepMode And Not stepTaken))

        gCurrentIter = gCurrentIter + 1
        
        OP1 = CStr(arrOp1(rowIdx + 1, 1))
        op2 = CStr(arrOp2(rowIdx + 1, 1))
        label = CStr(arrLabel(rowIdx + 1, 1))

        ' keep op2 uppercase for most uses except EQU literal semantics
        If UCase$(op2) <> "EQU" Then op2 = UCase$(op2)

        rowStat = CStr(arrRowStat(rowIdx + 1, 1))
        ' Check for code breaks; override in testrunner to have no breaks
        If rowStat = "B" And Not stepMode And Not skipBreaks Then
            If gBreak Then
                gBreak = False
            Else
                gBreak = True
                Exit Do
            End If
        End If

        executedThisIter = False

        ' Execute only if not compiled and opcode exists
        If rowStat <> "C" And opcode <> "" Then

            ' --- Trace memory sampling (before) ---
            memAddrHex = "": memBefore = "": memAfter = "": memNote = ""
            memTrackLen = 0
            memAddrDec = 0
            
            If TraceEnabled() Then
                ' Snapshot pre-state for address calculations
                preHL = ((myCpu.reg("H") And 255) * 256&) + (myCpu.reg("L") And 255)
                preSP = (myCpu.reg("SP") And &HFFFF&)
            
                Select Case opcode
            
                    ' ===== M-operand writes (HL indirect) =====
                    Case "MOV"
                        If UCase$(OP1) = "M" Then
                            memAddrDec = preHL
                            memTrackLen = 1
                        End If
            
                    Case "MVI", "INR", "DCR"
                        If UCase$(OP1) = "M" Then
                            memAddrDec = preHL
                            memTrackLen = 1
                        End If
            
                    ' ===== Direct memory writes =====
                    Case "STA"
                        If usrValIsHexString(OP1) Then
                            memAddrDec = usrHexToDec(OP1) And &HFFFF&
                            memTrackLen = 1
                        Else
                            memNote = "STA addr token=" & OP1
                        End If
            
                    Case "SHLD"
                        If usrValIsHexString(OP1) Then
                            memAddrDec = usrHexToDec(OP1) And &HFFFF&
                            memTrackLen = 2
                        Else
                            memNote = "SHLD addr token=" & OP1
                        End If
            
                    ' ===== Stack writes =====
                    Case "PUSH"
                        memAddrDec = (preSP - 2) And &HFFFF&
                        memTrackLen = 2
                        memNote = "PUSH writes SP-2..SP-1"
            
                    Case "CALL", "CC", "CNC", "CNZ", "CZ", "CP", "CM", "CPE", "CPO"
                        ' Conditional calls only write if taken; we’ll detect by before/after diff
                        memAddrDec = (preSP - 2) And &HFFFF&
                        memTrackLen = 2
                        memNote = "CALL writes SP-2..SP-1 (if taken)"
            
                    ' ===== XTHL swaps stack top =====
                    Case "XTHL"
                        memAddrDec = preSP
                        memTrackLen = 2
                        memNote = "XTHL swaps (SP),(SP+1)"
            
                End Select
                
                ' ===== Memory reads (optional) =====
                If memTrackLen = 0 And traceReads Then
                
                    Select Case opcode
                
                        ' MOV r, M  (read from (HL))
                        Case "MOV"
                            If UCase$(op2) = "M" Then
                                memAddrDec = preHL
                                memTrackLen = 1
                                memNote = "READ (HL)"
                            End If
                
                        ' ALU/LOGIC using M (read from (HL))
                        Case "ADD", "ADC", "SUB", "SBB", "ANA", "XRA", "ORA", "CMP"
                            If UCase$(OP1) = "M" Then
                                memAddrDec = preHL
                                memTrackLen = 1
                                memNote = "READ (HL)"
                            End If
                
                        ' Direct reads
                        Case "LDA"
                            If usrValIsHexString(OP1) Then
                                memAddrDec = usrHexToDec(OP1) And &HFFFF&
                                memTrackLen = 1
                                memNote = "READ (addr)"
                            Else
                                memNote = "READ LDA addr token=" & OP1
                            End If
                
                        Case "LHLD"
                            If usrValIsHexString(OP1) Then
                                memAddrDec = usrHexToDec(OP1) And &HFFFF&
                                memTrackLen = 2
                                memNote = "READ16 (addr)"
                            Else
                                memNote = "READ LHLD addr token=" & OP1
                            End If
                
                        ' Indirect reads via BC/DE
                        Case "LDAX"
                            If UCase$(OP1) = "B" Then
                                memAddrDec = ((myCpu.reg("B") And 255) * 256&) + (myCpu.reg("C") And 255)
                                memTrackLen = 1
                                memNote = "READ (BC)"
                            ElseIf UCase$(OP1) = "D" Then
                                memAddrDec = ((myCpu.reg("D") And 255) * 256&) + (myCpu.reg("E") And 255)
                                memTrackLen = 1
                                memNote = "READ (DE)"
                            End If
                
                        ' Stack reads
                        Case "POP"
                            memAddrDec = preSP
                            memTrackLen = 2
                            memNote = "READ16 (SP) POP"
                
                        Case "RET", "RNZ", "RZ", "RC", "RNC", "RP", "RM", "RPE", "RPO"
                            memAddrDec = preSP
                            memTrackLen = 2
                            memNote = "READ16 (SP) RET* (if taken)"
                    End Select
                
                End If
            
                ' Read BEFORE bytes if we have an address
                If memTrackLen = 1 Then
                    b0 = gMemory.addr(memAddrDec) And 255
                    memBefore = Right$("00" & hex$(b0), 2)
                    memAddrHex = Right$("0000" & hex$(memAddrDec), 4)
                ElseIf memTrackLen = 2 Then
                    b0 = gMemory.addr(memAddrDec) And 255
                    b1 = gMemory.addr((memAddrDec + 1) And &HFFFF&) And 255
                    memBefore = Right$("00" & hex$(b0), 2) & " " & Right$("00" & hex$(b1), 2)
                    memAddrHex = Right$("0000" & hex$(memAddrDec), 4)
                End If
            End If
            
            If debugMode Then
                RefreshFlags myCpu
                RefreshRegisters myCpu
                myCpu.RefreshStack True
                Application.Calculate
                Debug.Assert Not debugMode
            End If

            errcode = myCpu.RunOpcode(opcode, OP1, op2, label)
            executedThisIter = True
            
            If TraceEnabled() Then
                ' Read AFTER bytes for tracked address and include only if changed
                Dim isRead As Boolean
                If memTrackLen = 1 Then
                    a0 = gMemory.addr(memAddrDec) And 255
                    memAfter = Right$("00" & hex$(a0), 2)
            
                    isRead = (Left$(memNote, 4) = "READ")
                    
                    If isRead Then
                        ' Reads don't change memory; keep the read value visible.
                        memAfter = ""
                    Else
                        ' Writes: hide if no change
                        If memBefore = memAfter Then
                            memAddrHex = "": memBefore = "": memAfter = "": memNote = ""
                        End If
                    End If
                ElseIf memTrackLen = 2 Then
                    a0 = gMemory.addr(memAddrDec) And 255
                    a1 = gMemory.addr((memAddrDec + 1) And &HFFFF&) And 255
                    memAfter = Right$("00" & hex$(a0), 2) & " " & Right$("00" & hex$(a1), 2)
            
                    isRead = (Left$(memNote, 4) = "READ")
                    
                    If isRead Then
                        ' Reads don't change memory; keep the read value visible.
                        memAfter = ""
                    Else
                        ' Writes: hide if no change
                        If memBefore = memAfter Then
                            memAddrHex = "": memBefore = "": memAfter = "": memNote = ""
                        End If
                    End If
                End If
            
                ' Append trace row (post-state + optional memory delta)
                TraceAppend myCpu, pc_dec, opcode, OP1, op2, errcode, memAddrHex, memBefore, memAfter, memNote
            
                If stepMode And TraceLive() Then TraceFlush
            End If

            If errcode = ERR_EXEC_END Then
                endedByEnd = True
                Exit Do
            End If

            If errcode = ERR_BAD_OPCODE Then
                Range("Error").value = errcode
                Range("errMessage").value = "Error: Invalid OpCode: " & opcode & " at Line " & pc_dec
                Exit Do
            End If
            
            If gCurrentIter >= maxIters Then
                errcode = ERR_MAX_ITERS
                Range("Error").value = errcode
                Range("errMessage").value = "Error: Max Iterations Limit: " & gCurrentIter & " at Line " & pc_dec
                Exit Do
            End If

        End If

        ' Advance PC only after processing this row (consistent with your existing design)
        If errcode = 0 Then
            pc_dec = myCpu.IncPC()
        End If

        ' Update row index for next fetch
        rowIdx = myCpu.GetCurrentIdx
        If rowIdx < 0 Or rowIdx >= ubRows_Opcode Then Exit Do

        opcode = CStr(arrOpcode(rowIdx + 1, 1))

        ' In step mode, only count the step if an instruction executed
        If stepMode And executedThisIter Then
            stepTaken = True
        End If

        ' Paint UI in step mode
        If stepMode And errcode = 0 Then
            'advance to next non-blank row
            Do While opcode = "" And pc_dec <= memEnd
                pc_dec = myCpu.IncPC()
                rowIdx = pc_dec - line0_dec
                opcode = CStr(arrOpcode(rowIdx + 1, 1))
            Loop
            
            If myCpu.RegistersDirty Then
                RefreshRegisters myCpu
                myCpu.RegistersDirty = False
            End If
            If myCpu.FlagsDirty Then
                RefreshFlags myCpu
                myCpu.FlagsDirty = False
            End If
            If myCpu.StackDirty Then
                myCpu.RefreshStack True
                myCpu.StackDirty = False
            End If
        End If
        
        If Range("Stop").value = 1 Then
            errcode = ERR_STOP
            Range("Error").value = errcode
            Range("errMessage").value = "Error: Execution Stopped. Iteration: " & gCurrentIter & " at Line " & pc_dec
            Exit Do
        End If
    Loop

ExecFinalize:
    ' ========= Finalization =========
    If headless Or stepMode Or gBreak Then
        Application.Calculation = prevCalc
        Application.EnableEvents = True
        Application.ScreenUpdating = True

        If Not stepMode Then
            RefreshRegisters myCpu
            RefreshFlags myCpu
            myCpu.RefreshStack True
        Else
            Range("PC").value = usrHexToString(myCpu.reg("PC"))
        End If

        Application.StatusBar = False
        Range("consoleText") = usrForm.tbConsole.value
    End If

    If TraceEnabled() Then TraceFlush
    
    ' Show "Execution Complete" ONLY when END was encountered
    If gBreak Then
        Range("errMessage") = "Break at Line " & pc_dec
        Range("Reset").value = 0
    ElseIf endedByEnd Then
        Range("errMessage") = "Execution Complete"
    End If
End Sub

''' Component: Sheet3

''' Component: Compiler
Option Explicit

' --- Compile gating / caching ---
Private gCompileEnabled As Boolean
Private gCompiledValid As Boolean
Private gCompiledForResetStamp As Double   ' optional: timestamp of last reset compile
Private gLastCodeSignature As String       ' optional: detect edits without reset
' MemoryTable is 8 bytes wide: AB..AI
Private Const BYTES_PER_MEMROW As Long = 8
Private Const EMPTY_RUN_LIMIT As Long = 8
Private gMemTableCol As Long
Private pLabels As New clsLabels
' =============================================================================
' Assemble8080_ToMachine
' Compiles CPU listing into 8080 machine bytes and writes to:
'   - Column Z: base address for each 8-byte row
'   - MemoryTable (AB..AI): hex bytes
' Optionally writes "AA BB CC" into the listing Mem column and marks RowStat="C".
' =============================================================================
Public Sub Assemble8080_ToMachine()

    Dim wsCPU As Worksheet, wsOp As Worksheet
    Set wsCPU = ThisWorkbook.Worksheets("CPU")
    Set wsOp = ThisWorkbook.Worksheets("8080 Op to Hex") ' opcode mapping table
    Dim myMemory As clsMemory
    Dim lbl As clsLabelRecord
    
    usrMemReset
    ResetAddressList
    Set myMemory = gMemory
    ' Offsets into the CPU program grid (named ranges)
    Dim ofsLabel As Long, ofsOpcode As Long, ofsOp1 As Long, ofsOp2 As Long

    ofsLabel = CLng(wsCPU.Range("ofs_label").value)
    ofsOpcode = CLng(wsCPU.Range("ofs_opcode").value)
    ofsOp1 = CLng(wsCPU.Range("ofs_op1").value)
    ofsOp2 = CLng(wsCPU.Range("ofs_op2").value)
    gMemTableCol = CLng(wsCPU.Range("MemoryTableAddress").Column)
    

    ' Memory window
    Dim memStartDec As Long, memSize As Long
    memStartDec = usrHexToDec(CStr(wsCPU.Range("MemStart").value))
    memSize = usrHexToDec(CStr(wsCPU.Range("MemSize").value))

    ' Build opcode lookup dictionary from "8080 Op to Hex"
    Dim opMap As Object
    Set opMap = BuildOpcodeMap(wsOp)

    ' Labels/EQU/DB metadata (your existing class)
    Dim lbls As clsLabels
    Set lbls = New clsLabels

    ' MemoryTable byte cells (AB..AI) named range
    Dim rngMem As Range
    Set rngMem = wsCPU.Range("MemoryTable")

    Application.Calculate
    Application.ScreenUpdating = False

    ' Clear prior compiled bytes (only AB..AI)
    rngMem.ClearContents

    ' Also clear address column Z for the memory rows we might touch (overwrite is OK per you)
    wsCPU.Range(wsCPU.Cells(rngMem.row, gMemTableCol), _
                wsCPU.Cells(rngMem.row + rngMem.Rows.Count - 1, gMemTableCol)).ClearContents

    Dim base As Range
    Set base = wsCPU.Range("Line0") ' start of listing grid

    Dim pc As Long
    pc = memStartDec

    Dim r As Long, emptyRun As Long
    emptyRun = 0

    ' Walk listing rows (cap at memSize rows for safety)
    For r = 0 To memSize

        Dim cur As Range
        Set cur = base.offset(r, 0)

        Dim labelName As String, opcode As String, OP1 As String, op2 As String
        labelName = UCase$(Trim$(CStr(cur.offset(0, ofsLabel).value)))
        opcode = UCase$(Trim$(CStr(cur.offset(0, ofsOpcode).value)))
        OP1 = StripComment(Trim$(CStr(cur.offset(0, ofsOp1).value)))
        op2 = StripComment(Trim$(CStr(cur.offset(0, ofsOp2).value)))

        ' stop condition: consecutive empty program rows
        If (labelName = ";") Then
            GoTo NextRow
        End If
        
        If (labelName = "") And (opcode = "") And (OP1 = "") And (op2 = "") Then
            emptyRun = emptyRun + 1
            If emptyRun >= EMPTY_RUN_LIMIT Then Exit For
            GoTo NextRow
        Else
            emptyRun = 0
        End If

        Set lbl = pLabels.GetLabel(OP1)
        If Not lbl Is Nothing Then
            OP1 = lbl.clsAddressHex ' use the label's hex value/address
        End If

        Dim bytesOut As Variant
        bytesOut = Empty

        Select Case opcode
            Case ""
                GoTo NextRow

            Case "ORG"
                If OP1 = "" Then
                    RaiseAsmError wsCPU, "ORG missing operand", cur
                    Exit Sub
                End If
                pc = ResolveValue16(lbls, UCase$(OP1))
                GoTo NextRow

            Case "EQU"
                ' In your model, EQU constants are captured during label parsing already.
                GoTo NextRow

            Case "DB"
                ' String literal only for now (raw OP1)
                If Len(CStr(cur.offset(0, ofsOp1).value)) = 0 Then
                    RaiseAsmError wsCPU, "DB missing string", cur
                    Exit Sub
                End If
                bytesOut = StringToBytesVariant(CStr(cur.offset(0, ofsOp1).value))

            Case "DS"
                Dim ds As Long
                Dim i As Long
                If Not usrValIsHexString(cur.offset(0, ofsOp1).value) Then
                     RaiseAsmError wsCPU, "DS Invalid DS Length", cur
                    Exit Sub
                End If
                ds = usrHexToDec(cur.offset(0, ofsOp1).value)
                Dim out() As Variant
                ReDim out(0 To ds - 1)
                For i = 1 To ds
                    out(i - 1) = 255
                Next i
                bytesOut = out
                    
            Case Else
                ' Normal instruction: lookup + encode (1/2/3 bytes)
                Dim info As Variant
                info = HlpFindOpcode(opMap, opcode, UCase$(OP1), UCase$(op2))
                If IsEmpty(info) Then
                    RaiseAsmError wsCPU, "Unknown instruction form: " & opcode & " " & OP1 & " " & op2, cur
                    Exit Sub
                End If

                bytesOut = HlpEncodeInstruction(lbls, CByte(info(0)), CLng(info(1)), _
                                             CStr(info(2)), CStr(info(3)), _
                                             UCase$(OP1), UCase$(op2), opcode, cur, wsCPU)
        End Select

        ' Emit compiled bytes to MemoryTable using PC-based mapping (Z + AB..AI)
        If Not IsEmpty(bytesOut) Then
            EmitBytesToMemoryTable wsCPU, rngMem, memStartDec, pc, bytesOut, myMemory


            pc = pc + (UBound(bytesOut) - LBound(bytesOut) + 1)
        End If


NextRow:
    Next r
    Application.ScreenUpdating = True

    wsCPU.Range("errMessage").value = "Assemble complete"

End Sub

' =============================================================================
' Memory emit: PC -> row/col within AB..AI and base address into column Z
' =============================================================================
Private Sub EmitBytesToMemoryTable(ByVal wsCPU As Worksheet, _
 ByVal rngMem As Range, _
 ByVal memStartDec As Long, _
 ByVal startAddrDec As Long, _
 ByVal bytesOut As Variant, _
 ByRef myMem As clsMemory)

    Dim i As Long
    Dim baseAddr As Long
    Dim lastTargetRow As Long

    lastTargetRow = 0
    baseAddr = memStartDec

    For i = LBound(bytesOut) To UBound(bytesOut)

        Dim addr As Long
        addr = startAddrDec + (i - LBound(bytesOut))

        Dim offset As Long
        offset = addr - memStartDec
        If offset < 0 Then Exit Sub

        Dim rowIndex As Long, colIndex As Long
        rowIndex = offset \ BYTES_PER_MEMROW
        colIndex = offset Mod BYTES_PER_MEMROW

        Dim targetRow As Long, targetCol As Long
        targetRow = 1 + rowIndex
        targetCol = 1 + colIndex

        If targetRow > rngMem.Rows.Count Then Exit Sub
        If targetCol > rngMem.Columns.Count Then Exit Sub

        rngMem.Cells(targetRow, targetCol).value = Right$("00" & hex$(CLng(bytesOut(i)) And &HFF&), 2)
        myMem.addr(addr) = CLng(bytesOut(i)) And &HFF&

        baseAddr = memStartDec + (rowIndex * BYTES_PER_MEMROW)
        lastTargetRow = targetRow
    Next i

    If lastTargetRow > 0 Then
        wsCPU.Cells(rngMem.row + (lastTargetRow - 1), gMemTableCol).value = baseAddr
    End If

End Sub

' =============================================================================
' Opcode map from "8080 Op to Hex" table: key = MNEMONIC|OP1SPEC|OP2SPEC
' =============================================================================
Public Function BuildOpcodeMap(ByVal wsOp As Worksheet) As Object
    Dim dict As Object: Set dict = CreateObject("Scripting.Dictionary")
    dict.CompareMode = vbTextCompare

    Dim lastRow As Long
    lastRow = wsOp.Cells(wsOp.Rows.Count, 1).End(xlUp).row

    Dim r As Long, emptyRun As Long
    emptyRun = 0

    ' Columns in your sheet:
    ' A Opcode, B Hex, D OP1, E OP2, F Bytes
    For r = 2 To lastRow
        Dim mnem As String
        mnem = UCase$(Trim$(CStr(wsOp.Cells(r, 1).value)))
        If mnem = "" Then
            emptyRun = emptyRun + 1
            If emptyRun >= 10 Then Exit For
            GoTo ContinueRow
        Else
            emptyRun = 0
        End If

        Dim hexTxt As String: hexTxt = UCase$(Trim$(CStr(wsOp.Cells(r, 2).value)))
        Dim spec1 As String: spec1 = UCase$(Trim$(CStr(wsOp.Cells(r, 4).value)))
        Dim spec2 As String: spec2 = UCase$(Trim$(CStr(wsOp.Cells(r, 5).value)))
        Dim bytes As Long: bytes = CLng(wsOp.Cells(r, 6).value)

        If hexTxt = "" Or bytes < 1 Then GoTo ContinueRow

        Dim opByte As Long
        opByte = usrHexToDec(hexTxt) And &HFF&

        Dim key As String
        key = mnem & "|" & spec1 & "|" & spec2

        If Not dict.Exists(key) Then
            dict.ADD key, Array(opByte, bytes, spec1, spec2)
        End If

ContinueRow:
    Next r

    Set BuildOpcodeMap = dict
End Function

Private Function HlpFindOpcode(ByVal opMap As Object, ByVal mnem As String, ByVal OP1 As String, ByVal op2 As String) As Variant
    Dim key As String

    key = mnem & "|" & OP1 & "|" & op2
    If opMap.Exists(key) Then HlpFindOpcode = opMap(key): Exit Function

    key = mnem & "|" & OP1 & "|"
    If opMap.Exists(key) Then HlpFindOpcode = opMap(key): Exit Function

    Dim p As Variant
    For Each p In Array("BYTE", "ADDRESS", "PORT", "DATA")
        key = mnem & "|" & OP1 & "|" & CStr(p)
        If opMap.Exists(key) Then HlpFindOpcode = opMap(key): Exit Function
    Next p

    For Each p In Array("BYTE", "ADDRESS", "PORT", "DATA")
        key = mnem & "|" & CStr(p) & "|" & op2
        If opMap.Exists(key) Then HlpFindOpcode = opMap(key): Exit Function
        key = mnem & "|" & CStr(p) & "|"
        If opMap.Exists(key) Then HlpFindOpcode = opMap(key): Exit Function
        key = mnem & "|" & CStr(p) & "|" & CStr(p)
        If opMap.Exists(key) Then HlpFindOpcode = opMap(key): Exit Function
    Next p

    HlpFindOpcode = Empty
End Function

' =============================================================================
' Encode instruction bytes (1/2/3 bytes), using label/EQU resolution via clsLabels
' =============================================================================
Private Function HlpEncodeInstruction(ByVal lbls As clsLabels, ByVal opByte As Byte, ByVal nBytes As Long, _
                                   ByVal spec1 As String, ByVal spec2 As String, _
                                   ByVal OP1 As String, ByVal op2 As String, _
                                   ByVal opname As String, ByVal curRow As Range, ByVal wsCPU As Worksheet) As Variant

    Dim out() As Variant

    Select Case nBytes
        Case 1
            ReDim out(0 To 0)
            out(0) = opByte

        Case 2
            ReDim out(0 To 1)
            out(0) = opByte

            Dim imm8 As Long
            imm8 = ResolveValue8(lbls, HlpPickImmediateToken(spec1, spec2, OP1, op2), opname, wsCPU, curRow)
            out(1) = CByte(imm8 And &HFF&)

        Case 3
            ReDim out(0 To 2)
            out(0) = opByte

            Dim imm16 As Long
            imm16 = ResolveValue16(lbls, HlpPickImmediateToken(spec1, spec2, OP1, op2))

            ' 8080 immediates are little-endian: low byte then high byte
            out(1) = CByte(imm16 And &HFF&)
            out(2) = CByte((imm16 \ 256) And &HFF&)

        Case Else
            RaiseAsmError wsCPU, "Unsupported instruction length: " & nBytes & " for " & opname, curRow
            HlpEncodeInstruction = Empty
            Exit Function
    End Select

    HlpEncodeInstruction = out
End Function

Private Function HlpPickImmediateToken(ByVal spec1 As String, ByVal spec2 As String, ByVal OP1 As String, ByVal op2 As String) As String
    If HlpIsPlaceholder(spec1) Then
        HlpPickImmediateToken = OP1
    ElseIf HlpIsPlaceholder(spec2) Then
        HlpPickImmediateToken = op2
    Else
        If Len(op2) > 0 Then HlpPickImmediateToken = op2 Else HlpPickImmediateToken = OP1
    End If
End Function

Private Function HlpIsPlaceholder(ByVal s As String) As Boolean
    s = UCase$(Trim$(s))
    HlpIsPlaceholder = (s = "BYTE" Or s = "ADDRESS" Or s = "PORT" Or s = "DATA")
End Function

' =============================================================================
' Operand resolution
' =============================================================================
Private Function ResolveValue8(ByVal lbls As clsLabels, ByVal token As String, _
                               ByVal opname As String, ByVal wsCPU As Worksheet, ByVal curRow As Range) As Long
    token = UCase$(Trim$(token))
    If token = "" Then
        RaiseAsmError wsCPU, opname & " missing 8-bit immediate", curRow
        ResolveValue8 = 0
        Exit Function
    End If

    Dim rec As clsLabelRecord
    Set rec = lbls.GetLabel(token)
    If Not rec Is Nothing Then
        ResolveValue8 = (rec.clsAddress And &HFF&)
        Exit Function
    End If

    If usrValIsHexString(token) Then
        ResolveValue8 = usrHexToDec(token) And &HFF&
    ElseIf IsNumeric(token) Then
        ResolveValue8 = CLng(token) And &HFF&
    Else
        RaiseAsmError wsCPU, opname & " invalid imm8: " & token, curRow
        ResolveValue8 = 0
    End If
End Function

Private Function ResolveValue16(ByVal lbls As clsLabels, ByVal token As String) As Long
    token = UCase$(Trim$(token))
    If token = "" Then ResolveValue16 = 0: Exit Function

    Dim rec As clsLabelRecord
    Set rec = lbls.GetLabel(token)
    If Not rec Is Nothing Then
        ResolveValue16 = (rec.clsAddress And &HFFFF&)
        Exit Function
    End If

    If usrValIsHexString(token) Then
        ResolveValue16 = usrHexToDec(token) And &HFFFF&
    ElseIf IsNumeric(token) Then
        ResolveValue16 = CLng(token) And &HFFFF&
    Else
        ResolveValue16 = 0
    End If
End Function

' =============================================================================
' DB string literal only (for now)
' =============================================================================
Private Function StringToBytesVariant(ByVal s As String) As Variant
    Dim out() As Variant, i As Long
    If Len(s) = 0 Then
        StringToBytesVariant = Empty
        Exit Function
    End If
    ReDim out(0 To Len(s))
    For i = 1 To Len(s)
        out(i - 1) = Asc(Mid$(s, i, 1)) And &HFF&
    Next i
    out(Len(s)) = 0
    StringToBytesVariant = out
End Function

Private Function BytesToHexString(ByVal bytesOut As Variant) As String
    Dim i As Long, s As String
    s = ""
    For i = LBound(bytesOut) To UBound(bytesOut)
        s = s & IIf(Len(s) > 0, " ", "") & Right$("00" & hex$(CLng(bytesOut(i)) And &HFF&), 2)
    Next i
    BytesToHexString = s
End Function

' =============================================================================
' Helpers / safety wrappers

Private Function StripComment(ByVal s As String) As String
    Dim p As Long
    p = InStr(1, s, ";", vbBinaryCompare)
    If p > 0 Then StripComment = Trim$(Left$(s, p - 1)) Else StripComment = Trim$(s)
End Function

' Class:        clsMemory
' Purpose:      Memory representation

'================================================================================
Private pStartAddress As Long
Private pMemSize As Long
Private pInitialized As Boolean
Private pMemory() As Byte
Private pMemError As Boolean

Private Sub Class_Initialize()
    Dim wsCPU As Worksheet
    Set wsCPU = ThisWorkbook.Worksheets("CPU")
    pStartAddress = usrHexToDec(CStr(wsCPU.Range("MemStart").value))
    pMemSize = usrHexToDec(CStr(wsCPU.Range("MemSize").value))
    ReDim pMemory(0 To pMemSize)
    pInitialized = True
    pMemError = False
End Sub

Public Function IsInitialized() As Boolean
    IsInitialized = pInitialized
End Function

Public Property Get addr(ByVal memAddr As Long) As Byte
    Dim offset As Long
    offset = memAddr - pStartAddress
    If offset < 0 Or offset > pMemSize Then
        addr = 0
        pMemError = True
    Else
        addr = pMemory(offset)
        pMemError = False
    End If
End Property

Public Property Let addr(ByVal memAddr As Long, ByVal value As Byte)
    Dim offset As Long
    offset = memAddr - pStartAddress
    If offset < 0 Or offset > pMemSize Then
        Debug.Print "Invalid Address: " & offset & " non in range (0," & pMemSize & ")"
        pMemError = False
        Exit Property
    End If
    pMemError = False
    pMemory(offset) = value
End Property

Public Function StartAddress() As Long
    StartAddress = pStartAddress
End Function

Public Function MemorySize() As Long
    MemorySize = pMemSize
End Function
Public Property Get MemError() As Boolean
    MemError = pMemError
End Property

' Class:        clsAddrRecord
' Purpose:      Node in a double linked list used for memory addressing
'================================================================================
Public clsAddress As Long
Public clsAddressHex As String
Public clsAddressLast As Long
Public clsAddressNext As Long
Public clsIndex As Long

''' Component: clsAddrList
Option Explicit
'================================================================================
' Class:        clsAddrList
'================================================================================

Private pNodes As Object
Private wsCPU As Worksheet

Private Sub Class_Initialize()
    Set wsCPU = ThisWorkbook.Worksheets("CPU")
    Set pNodes = CreateObject("Scripting.Dictionary")
    ParseAddresses
End Sub

Public Sub ParseAddresses()
    Dim data As Variant
    Dim r As Long
    Dim nodeRec As clsAddrRecord
    Dim line0 As Range
    Dim maxRows As Long
    Dim addr_hex As String
    Dim addr_dec As Long
    Dim addr_previous As Long
    Dim emptyRun As Long
    Dim ofs_opcode As Long
    Dim ofs_label As Long
    Dim opcode As String
    Const EMPTY_RUN_LIMIT As Long = 8   ' stop after 8 consecutive empty rows
  
    pNodes.RemoveAll


    Set line0 = wsCPU.Range("Line0")
    ofs_opcode = wsCPU.Range("ofs_opcode").value
    ofs_label = wsCPU.Range("ofs_label").value
    addr_previous = 0
    emptyRun = 0

    ' Offsets into the CPU grid
    ' Upper bound so we don't scan forever: MemSize + 1 is perfect
    ' (Your grid is MemStart..MemEnd; MemSize is in hex on CPU sheet.)
    maxRows = CLng(usrHexToDec(CStr(wsCPU.Range("MemSize").value))) + 1
    If maxRows < 1 Then maxRows = 1
    If maxRows > 5000 Then maxRows = 5000 ' hard safety cap

    ' Pull a block once for speed: from column 1 through ofs_ops2
    data = line0.offset(0, 0).Resize(maxRows, ofs_opcode + 1).value
    For r = 1 To UBound(data, 1)
        ' Read the program fields from the array (1-based array)
        addr_dec = CLng(data(r, 1))
        opcode = CStr(data(r, ofs_opcode + 1))
        'skip comments
        If data(r, ofs_label + 1) = ";" Then
            GoTo NextRow
        End If
        
        ' Determine whether this row is "program-empty"
        If data(r, 1) = "" Or addr_dec = addr_previous Or opcode = "" Then
            emptyRun = emptyRun + 1
            If emptyRun >= EMPTY_RUN_LIMIT Then Exit For
            GoTo NextRow
        Else
            emptyRun = 0
        End If
        
        If opcode = "EQU" Or opcode = "ORG" Then
            GoTo NextRow
        End If
        
        If data(r, 1) = "" Or addr_dec = 0 Then
            Exit For
        End If
        Set nodeRec = New clsAddrRecord
        nodeRec.clsAddress = addr_dec
        nodeRec.clsAddressHex = usrHexToString(addr_dec)
        If r = 1 Then
            nodeRec.clsAddressLast = 0
        Else
            nodeRec.clsAddressLast = addr_previous
        End If
        If r = maxRows Then
            nodeRec.clsAddressNext = 0
        Else
            nodeRec.clsAddressNext = CLng(data(r + 1, 1))
        End If
        nodeRec.clsIndex = r - 1
        pNodes.ADD addr_dec, nodeRec
        addr_previous = addr_dec

NextRow:
    Next r
End Sub

'------------------------------------------------------------------------------
' GetLabel
' Returns the label record if found, else Nothing.
'------------------------------------------------------------------------------
Public Function GetAddr(ByVal addr As Long) As clsAddrRecord
    If pNodes.Exists(addr) Then
        Set GetAddr = pNodes(addr)
    Else
        Set GetAddr = Nothing
    End If
End Function

Public Function Size() As Long
    Size = pNodes.Count
End Function

''' Component: Trace
Option Explicit

'=========================
' Trace mode (buffered)
'=========================
Private Const TRACE_BATCH As Long = 500          ' grow buffer in chunks
Private Const TRACE_MAX_ROWS As Long = 20000     ' hard cap safety

Private gTraceArr() As Variant                  ' 2D array buffer
Private gTraceCount As Long                     ' total entries captured
Private gTraceWritten As Long                   ' how many entries flushed to sheet
Private gHeadersWritten As Boolean

' Columns (fixed order)
'  1 Step#
'  2 PC
'  3 Op
'  4 OP1
'  5 OP2
'  6 A
'  7 B
'  8 C
'  9 D
' 10 E
' 11 H
' 12 L
' 13 SP
' 14 S
' 15 Z
' 16 P
' 17 CY
' 18 AC
' 19 Err
'20 memAddr
'21 MemBefore
'22 MemAfter
'23 MemNote
Private Const TRACE_COLS As Long = 23

Public Function TraceEnabled() As Boolean
    ' Controlled by CPU!Trace (named range). Missing range => False.
    On Error Resume Next
    TraceEnabled = (CLng(ThisWorkbook.Worksheets("CPU").Range("Trace").value) = 1)
    If Err.Number <> 0 Then TraceEnabled = False
    On Error GoTo 0
End Function

Public Function TraceLive() As Boolean
    ' Optional CPU!TraceLive (named range). Missing => False.
    On Error Resume Next
    TraceLive = (CLng(ThisWorkbook.Worksheets("CPU").Range("TraceLive").value) = 1)
    If Err.Number <> 0 Then TraceLive = False
    On Error GoTo 0
End Function
Public Function TraceMemReads() As Boolean
    On Error Resume Next
    TraceMemReads = (CLng(ThisWorkbook.Worksheets("CPU").Range("TraceMemReads").value) = 1)
    If Err.Number <> 0 Then TraceMemReads = False
    On Error GoTo 0
End Function
Public Sub TraceClear()
    gTraceCount = 0
    gTraceWritten = 0
    gHeadersWritten = False
    Erase gTraceArr

    Dim ws As Worksheet
    Set ws = TraceGetSheet(True)
    ws.Cells.Clear
End Sub

Private Function TraceGetSheet(Optional ByVal createIfMissing As Boolean = True) As Worksheet
    On Error Resume Next
    Set TraceGetSheet = ThisWorkbook.Worksheets("Trace")
    On Error GoTo 0

    If TraceGetSheet Is Nothing And createIfMissing Then
        Set TraceGetSheet = ThisWorkbook.Worksheets.ADD(After:=ThisWorkbook.Sheets(ThisWorkbook.Sheets.Count))
        TraceGetSheet.name = "Trace"
    End If
End Function

Private Sub TraceEnsureHeaders()
    If gHeadersWritten Then Exit Sub

    Dim ws As Worksheet
    Set ws = TraceGetSheet(True)

    Dim headers As Variant
    headers = Array("Step", "PC", "Op", "OP1", "OP2", _
                    "A", "B", "C", "D", "E", "H", "L", "SP", _
                    "S", "Z", "P", "CY", "AC", "Err", _
                    "MemAddr", "MemBefore", "MemAfter", "MemNote")

    ws.Range("A1").Resize(1, UBound(headers) + 1).value = headers
    ws.Rows(1).Font.Bold = True

    ws.Columns("A").ColumnWidth = 8
    ws.Columns("B").ColumnWidth = 6
    ws.Columns("C").ColumnWidth = 8
    ws.Columns("D:E").ColumnWidth = 10
    ws.Columns("F:M").ColumnWidth = 4
    ws.Columns("N:S").ColumnWidth = 4
    ws.Columns("T").ColumnWidth = 8   ' MemAddr
    ws.Columns("U:V").ColumnWidth = 10 ' before/after
    ws.Columns("W").ColumnWidth = 28  ' note
    ws.Columns("T:W").NumberFormat = "@"
    
    ' Keep hex-looking fields as text (prevents Excel auto-formatting)
    ws.Columns("B").NumberFormat = "@"
    ws.Columns("F:M").NumberFormat = "@"

    gHeadersWritten = True
End Sub

Private Sub TraceEnsureCapacity(ByVal neededRow As Long)
    Dim curRows As Long

    ' If the array is not allocated yet, UBound will error.
    On Error Resume Next
    curRows = UBound(gTraceArr, 1)
    If Err.Number <> 0 Then
        Err.Clear
        ReDim gTraceArr(1 To TRACE_BATCH, 1 To TRACE_COLS)
        Exit Sub
    End If
    On Error GoTo 0

    If neededRow <= curRows Then Exit Sub

    ReDim Preserve gTraceArr(1 To (curRows + TRACE_BATCH), 1 To TRACE_COLS)
End Sub

Private Function Hex2(ByVal v As Long) As String
    Hex2 = Right$("00" & hex$(v And 255), 2)
End Function

Private Function Hex4(ByVal v As Long) As String
    Hex4 = Right$("0000" & hex$(v And &HFFFF&), 4)
End Function

Public Sub TraceAppend(ByRef cpu As clsDecCPU, _
                       ByVal pc_dec As Long, _
                       ByVal opcode As String, _
                       ByVal OP1 As String, _
                       ByVal op2 As String, _
                       ByVal errcode As Long, _
                       Optional ByVal memAddr As String = "", _
                       Optional ByVal memBefore As String = "", _
                       Optional ByVal memAfter As String = "", _
                       Optional ByVal memNote As String = "")
                       
    If Not TraceEnabled() Then Exit Sub
    If gTraceCount >= TRACE_MAX_ROWS Then Exit Sub

    TraceEnsureHeaders

    gTraceCount = gTraceCount + 1
    TraceEnsureCapacity gTraceCount

    Dim r As Long: r = gTraceCount

    gTraceArr(r, 1) = r
    gTraceArr(r, 2) = Hex4(pc_dec)
    gTraceArr(r, 3) = opcode
    gTraceArr(r, 4) = OP1
    gTraceArr(r, 5) = op2

    gTraceArr(r, 6) = Hex2(cpu.reg("A"))
    gTraceArr(r, 7) = Hex2(cpu.reg("B"))
    gTraceArr(r, 8) = Hex2(cpu.reg("C"))
    gTraceArr(r, 9) = Hex2(cpu.reg("D"))
    gTraceArr(r, 10) = Hex2(cpu.reg("E"))
    gTraceArr(r, 11) = Hex2(cpu.reg("H"))
    gTraceArr(r, 12) = Hex2(cpu.reg("L"))
    gTraceArr(r, 13) = Hex4(cpu.reg("SP"))

    gTraceArr(r, 14) = cpu.Flag("Sign")
    gTraceArr(r, 15) = cpu.Flag("Zero")
    gTraceArr(r, 16) = cpu.Flag("Parity")
    gTraceArr(r, 17) = cpu.Flag("Carry")
    gTraceArr(r, 18) = cpu.Flag("AC")

    gTraceArr(r, 19) = errcode
    
    gTraceArr(r, 20) = memAddr
    gTraceArr(r, 21) = memBefore
    gTraceArr(r, 22) = memAfter
    gTraceArr(r, 23) = memNote
End Sub

Public Sub TraceFlush(Optional ByVal forceAll As Boolean = False)
    If Not TraceEnabled() Then Exit Sub
    If gTraceCount = 0 Then Exit Sub

    TraceEnsureHeaders

    Dim ws As Worksheet
    Set ws = TraceGetSheet(True)

    Dim startRow As Long, rowsToWrite As Long
    If forceAll Then
        gTraceWritten = 0
    End If

    rowsToWrite = gTraceCount - gTraceWritten
    If rowsToWrite <= 0 Then Exit Sub

    startRow = 2 + gTraceWritten

    ws.Range("A" & startRow).Resize(rowsToWrite, TRACE_COLS).value = _
        wsParentSlice(gTraceArr, gTraceWritten + 1, gTraceCount, TRACE_COLS)

    gTraceWritten = gTraceCount
End Sub

' Helper: slice a 2D array [r1..r2, 1..cols] into a new 2D array
Private Function wsParentSlice(ByRef src As Variant, ByVal r1 As Long, ByVal r2 As Long, ByVal cols As Long) As Variant
    Dim out As Variant
    Dim rr As Long, cc As Long, rOut As Long
    ReDim out(1 To (r2 - r1 + 1), 1 To cols)
    rOut = 0
    For rr = r1 To r2
        rOut = rOut + 1
        For cc = 1 To cols
            out(rOut, cc) = src(rr, cc)
        Next cc
    Next rr
    wsParentSlice = out
End Function

' Convenience button macros
Public Sub BTN_ClearTrace()
    TraceClear
End Sub

Public Sub BTN_FlushTrace()
    TraceFlush True
End Sub
